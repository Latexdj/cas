// Get academic years and semesters with auto-select functionality
function getAcademicYearsAndSemesters() {
  try {
    const academicYearSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('AcademicYear');
    const lastRow = academicYearSheet.getLastRow();
    
    if (lastRow < 2) {
      return { years: [], currentYear: '', currentSemester: '' };
    }
    
    // Get all data from AcademicYear sheet (Columns A, B, C, D)
    const data = academicYearSheet.getRange(2, 1, lastRow - 1, 4).getValues();
    
    const years = [];
    let currentYear = '';
    let currentSemester = '';
    
    data.forEach(row => {
      const year = row[0];
      const yearStatus = row[1] ? row[1].toString().trim().toLowerCase() : '';
      const semester = row[2];
      const semesterStatus = row[3] ? row[3].toString().trim().toLowerCase() : '';
      
      // Add year to the list if it exists
      if (year && year !== '') {
        years.push(year);
        
        // Check if this is the current academic year
        if (yearStatus === 'current') {
          currentYear = year;
          
          // If we found the current year, also check for current semester in this row
          if (semesterStatus === 'current') {
            currentSemester = convertSemesterToNumber(semester);
          }
        }
      }
    });
    
    Logger.log('Academic years found: ' + years.join(', '));
    Logger.log('Current academic year: ' + currentYear);
    Logger.log('Current semester: ' + currentSemester);
    
    return {
      years: years,
      currentYear: currentYear,
      currentSemester: currentSemester
    };
  } catch (error) {
    Logger.log('Error getting academic years and semesters: ' + error.message);
    return { years: [], currentYear: '', currentSemester: '' };
  }
}

// Helper function to convert "Semester 1" to "1" and "Semester 2" to "2"
function convertSemesterToNumber(semesterText) {
  if (!semesterText) return '';
  
  const semesterStr = semesterText.toString().trim().toLowerCase();
  
  if (semesterStr.includes('1') || semesterStr === 'semester 1') {
    return '1';
  } else if (semesterStr.includes('2') || semesterStr === 'semester 2') {
    return '2';
  } else if (semesterStr === '1' || semesterStr === '2') {
    return semesterStr;
  }
  
  return '';
}

// Get all available teaching locations from Classes sheet
function getTeachingLocations() {
  try {
    const classesSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Classes');
    const lastRow = classesSheet.getLastRow();
    
    if (lastRow < 2) {
      return [];
    }
    
    // Check if we have the extended structure (with coordinates)
    const headerRow = classesSheet.getRange(1, 1, 1, classesSheet.getLastColumn()).getValues()[0];
    const hasExtendedStructure = headerRow.some(cell => 
      cell && cell.toString().toLowerCase().includes('latitude')
    );
    
    let data;
    if (hasExtendedStructure) {
      // Use extended structure with coordinates
      data = classesSheet.getRange(2, 1, lastRow - 1, 5).getValues();
    } else {
      // Use basic structure (only class names)
      data = classesSheet.getRange(2, 1, lastRow - 1, 1).getValues();
    }
    
    const locations = [];
    
    data.forEach(row => {
      const className = row[0];
      
      if (hasExtendedStructure) {
        const locationType = row[1] || 'Classroom';
        const latitude = row[2];
        const longitude = row[3];
        const radius = row[4] || 30;
        
        if (className && latitude && longitude) {
          locations.push({
            name: className,
            type: locationType,
            latitude: latitude,
            longitude: longitude,
            radius: radius,
            hasCoordinates: true
          });
        } else if (className) {
          // Location without coordinates
          locations.push({
            name: className,
            type: locationType,
            hasCoordinates: false
          });
        }
      } else {
        // Basic structure - just class names
        if (className) {
          locations.push({
            name: className,
            type: 'Classroom',
            hasCoordinates: false
          });
        }
      }
    });
    
    // Add special locations if they don't exist
    const specialLocations = [
      { name: 'Catering Hall', type: 'CateringHall', hasCoordinates: false },
      { name: 'Library', type: 'Library', hasCoordinates: false },
      { name: 'ICT Lab', type: 'ICTLab', hasCoordinates: false }
    ];
    
    specialLocations.forEach(specialLoc => {
      if (!locations.find(loc => loc.name === specialLoc.name)) {
        locations.push(specialLoc);
      }
    });
    
    Logger.log('Found ' + locations.length + ' teaching locations');
    return locations;
  } catch (error) {
    Logger.log('Error getting teaching locations: ' + error.message);
    return [];
  }
}

// Verify if teacher is in the selected teaching location
function verifyTeachingLocation(selectedLocation, userLat, userLng) {
  try {
    const locations = getTeachingLocations();
    const location = locations.find(loc => loc.name === selectedLocation);
    
    if (!location) {
      return {
        valid: true, // Don't block submission if location not found
        message: 'Location not configured for verification',
        distance: 0,
        verified: false
      };
    }
    
    if (!location.hasCoordinates) {
      return {
        valid: true, // Don't block submission if no coordinates
        message: 'Location coordinates not configured',
        distance: 0,
        verified: false
      };
    }
    
    const distance = calculateDistance(userLat, userLng, location.latitude, location.longitude);
    const isInRange = distance <= location.radius;
    
    if (isInRange) {
      return {
        valid: true,
        message: `✓ You are in ${selectedLocation} (${Math.round(distance)}m from center)`,
        distance: distance,
        verified: true
      };
    } else {
      return {
        valid: false,
        message: `✗ You are ${Math.round(distance)}m away from ${selectedLocation} (max ${location.radius}m allowed)`,
        distance: distance,
        verified: true
      };
    }
    
  } catch (error) {
    Logger.log('Error verifying teaching location: ' + error.message);
    return {
      valid: true, // Don't block submission on error
      message: 'Location verification error: ' + error.message,
      distance: 0,
      verified: false
    };
  }
}

// Calculate distance between two coordinates using Haversine formula
function calculateDistance(lat1, lon1, lat2, lon2) {
  const R = 6371000; // Earth's radius in meters
  const dLat = (lat2 - lat1) * Math.PI / 180;
  const dLon = (lon2 - lon1) * Math.PI / 180;
  const a = 
    Math.sin(dLat/2) * Math.sin(dLat/2) +
    Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * 
    Math.sin(dLon/2) * Math.sin(dLon/2);
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
  const distance = R * c;
  return distance;
}

// Get teacher's schedule for today
function getTeacherSchedule(teacherName) {
  try {
    const timetableSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Timetable');
    const lastRow = timetableSheet.getLastRow();
    if (lastRow < 2) return [];

    const timetableData = timetableSheet.getRange(2, 1, lastRow - 1, 6).getValues();

    const today = getGhanaDate();
    const todayDay = today.getDay();

    Logger.log('Today day number: ' + todayDay);
    Logger.log('TeacherName received: "' + teacherName + '"');

    const todaysClasses = timetableData.filter(row => {
      const day = Number(row[0]);
      const teacher = row[3] ? row[3].toString().trim() : '';
      const searchTeacher = teacherName ? teacherName.toString().trim() : '';
      
      return day === todayDay && teacher === searchTeacher;
    });

    Logger.log(`Found ${todaysClasses.length} classes for today`);

    const schedule = todaysClasses.map(row => {
      const startTime = formatTime(row[1]);
      const endTime = formatTime(row[2]);
      
      const start = new Date(`1970-01-01T${startTime}`);
      const end = new Date(`1970-01-01T${endTime}`);
      const durationMinutes = (end - start) / (1000 * 60);
      const periods = Math.round(durationMinutes / 60);

      return {
        className: row[5],
        subject: row[4],
        startTime: startTime,
        endTime: endTime,
        periods: periods
      };
    });

    return schedule;
  } catch (error) {
    Logger.log('Error getting teacher schedule: ' + error.message);
    Logger.log('Error stack: ' + error.stack);
    return [];
  }
}

// Helper function to format time from Google Sheets date/time formats
function formatTime(dateValue) {
  try {
    if (!dateValue) return '';

    // If already a Date object
    if (dateValue instanceof Date) {
      return Utilities.formatDate(dateValue, Session.getScriptTimeZone(), 'HH:mm');
    }

    // If it's a string (ISO format or plain time)
    if (typeof dateValue === 'string') {
      // Handle ISO or timestamp-like strings (e.g., "2025-11-03T08:45:00Z")
      if (dateValue.includes('T')) {
        const date = new Date(dateValue);
        if (!isNaN(date.getTime())) {
          return Utilities.formatDate(date, Session.getScriptTimeZone(), 'HH:mm');
        }
      }

      // Handle plain time string (e.g., "08:45")
      if (/^\d{1,2}:\d{2}(:\d{2})?$/.test(dateValue)) {
        return dateValue.slice(0, 5); // "08:45" → "08:45"
      }

      return '';
    }

    // If it's a serial number from Google Sheets
    if (typeof dateValue === 'number') {
      const date = new Date((dateValue - 25569) * 86400 * 1000);
      return Utilities.formatDate(date, Session.getScriptTimeZone(), 'HH:mm');
    }

    return '';
  } catch (e) {
    Logger.log('Error formatting time: ' + e.message);
    return '';
  }
}

// Main attendance submission function
function submitData(data) {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');

    // Get the current date and time
    const timeZone = SpreadsheetApp.getActiveSpreadsheet().getSpreadsheetTimeZone();
    const now = new Date();
    const date = Utilities.formatDate(now, timeZone, 'yyyy-MM-dd');
    const time = Utilities.formatDate(now, timeZone, 'HH:mm:ss');
    
    // Calculate week number
    const weekNumber = getWeekNumber(now);

    // Validate academic year and semester
    let academicYear = data.academicYear;
    let semester = data.semester;
    
    if (!academicYear) {
      academicYear = getCurrentAcademicYear();
      if (!academicYear) {
        throw new Error('Academic year is required');
      }
    }
    
    if (!semester) {
      semester = getCurrentSemester();
      if (!semester) {
        throw new Error('Semester is required');
      }
    }

    const name = data.name;
    const subject = data.subject;
    const classNames = data.classNames;
    const periods = data.periods;
    const topic = data.topic;
    const gps = data.gps;
    const imageBase64 = data.imageBase64;
    const teachingLocation = data.teachingLocation || 'Not specified';

    // Extract GPS coordinates for verification
    let locationVerification = { valid: true, message: 'Location not verified', verified: false };
    
    if (gps && gps.includes(',')) {
      const gpsCoords = gps.split(',');
      if (gpsCoords.length === 2) {
        const userLat = parseFloat(gpsCoords[0]);
        const userLng = parseFloat(gpsCoords[1]);
        
        // Verify teaching location if selected
        if (teachingLocation && teachingLocation !== 'Not specified') {
          locationVerification = verifyTeachingLocation(teachingLocation, userLat, userLng);
          
          if (!locationVerification.valid) {
            throw new Error(`Location verification failed:\n${locationVerification.message}`);
          }
        }
      }
    }

    // Get all data from the sheet, skipping header row
    const lastRow = sheet.getLastRow();

    if (lastRow > 1) {
      const dataRange = sheet.getRange(2, 1, lastRow - 1, 7);
      const values = dataRange.getValues();

      // Validate duplicate entry for the same name, subject, overlapping class(es), and date
      for (let i = 0; i < values.length; i++) {
        const recordedDate = Utilities.formatDate(new Date(values[i][0]), timeZone, 'yyyy-MM-dd');
        const recordedName = values[i][4];
        const recordedSubject = values[i][5];
        const recordedClasses = values[i][6];

        if (recordedDate === date && recordedName === name && recordedSubject === subject) {
          const recordedClassArray = recordedClasses.split(', ').map(c => c.trim());
          const currentClassArray = classNames.split(', ').map(c => c.trim());

          if (currentClassArray.some(cls => recordedClassArray.includes(cls))) {
            throw new Error(
              `Attendance already recorded for ${name} in overlapping class(es) (${recordedClasses}) for this subject today.`
            );
          }
        }
      }
    }

    // If no duplicates are found, proceed to save the image and append the data
    const folder = DriveApp.getFolderById('1ACTWbfX3eDToOxmoVbh7TLcwuTCHCI4k');
    const fileName = name + '_' + date + '_' + classNames.replace(', ', '_') + '_' + now.getTime() + '.png';
    const imageBlob = Utilities.newBlob(Utilities.base64Decode(imageBase64.split(',')[1]), 'image/png', fileName);
    const file = folder.createFile(imageBlob);
    const pictureUrl = file.getUrl();

    // Append the data to the Google Sheet (updated with teaching location and verification)
    sheet.appendRow([
      date, time, academicYear, semester, name, 
      subject, classNames, periods, topic, 
      gps, pictureUrl, weekNumber, teachingLocation, locationVerification.message
    ]);

    // Get the last row (newly added) and set the URL as a chip
    const newRow = sheet.getLastRow();
    const urlCell = sheet.getRange(newRow, 11); // Column 11 for photo URL
    urlCell.setFormula('=HYPERLINK("' + pictureUrl + '", "View Photo")');

    let successMessage = 'Attendance recorded successfully!';
    if (teachingLocation && teachingLocation !== 'Not specified') {
      successMessage += `\n\nLocation: ${teachingLocation}\n${locationVerification.message}`;
    }

    return successMessage;
    
  } catch (error) {
    Logger.log('Error in submitData: ' + error.message);
    throw new Error('Error: ' + error.message);
  }
}

// Week number calculation
function getWeekNumber(date) {
  const thursday = new Date(date);
  thursday.setDate(date.getDate() + (4 - date.getDay()));
  
  const jan4 = new Date(thursday.getFullYear(), 0, 4);
  
  const weekNumber = 1 + Math.round(((thursday.getTime() - jan4.getTime()) / 86400000 - 3 + (jan4.getDay() + 6) % 7) / 7);
  
  return weekNumber;
}

// NEW: Check for absences once daily at 16:00 for the entire day (including weekends)
function checkForAbsencesDaily() {
  try {
    Logger.log('=== STARTING DAILY ABSENCE CHECK (4 PM) ===');
    
    const today = getGhanaDate();
    const dayOfWeek = today.getDay(); // 0=Sunday, 1=Monday, ..., 6=Saturday
    const dayName = getDayName(dayOfWeek);
    
    Logger.log(`Today is: ${dayName} (${dayOfWeek})`);
    
    const currentDate = Utilities.formatDate(today, 'Africa/Accra', 'yyyy-MM-dd');
    const currentTimeStr = getCurrentGhanaTime();
    
    Logger.log(`Current date: ${currentDate}, Time: ${currentTimeStr}`);
    
    // Get all lessons for today from timetable (works for any day including weekends)
    const todaysLessons = getTodaysLessons();
    Logger.log(`Found ${todaysLessons.length} scheduled lessons for ${dayName}`);
    
    if (todaysLessons.length === 0) {
      Logger.log(`No scheduled lessons found for ${dayName}`);
      return;
    }
    
    // Check each lesson for attendance
    let absenceCount = 0;
    let presentCount = 0;
    
    todaysLessons.forEach(lesson => {
      const isPresent = checkIfAttendanceSubmitted(lesson, today);
      
      if (!isPresent) {
        Logger.log(`DAILY ABSENCE: ${lesson.teacher} - ${lesson.subject} - ${lesson.className}`);
        markTeacherAbsentDaily(lesson, today, currentTimeStr);
        absenceCount++;
      } else {
        Logger.log(`PRESENT: ${lesson.teacher} - ${lesson.className}`);
        presentCount++;
      }
    });
    
    Logger.log(`=== DAILY ABSENCE CHECK COMPLETED ===`);
    Logger.log(`Summary: ${presentCount} present, ${absenceCount} absent out of ${todaysLessons.length} total lessons`);
    
  } catch (error) {
    Logger.log('ERROR in daily absence check: ' + error.toString());
  }
}

// Helper function to get day name
function getDayName(dayNumber) {
  const days = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'];
  return days[dayNumber];
}

// Get all lessons scheduled for today (works for any day including weekends)
function getTodaysLessons() {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const timetableSheet = ss.getSheetByName('Timetable');
    if (!timetableSheet) return [];

    const data = timetableSheet.getDataRange().getValues();
    const todaysLessons = [];

    // Use Ghana time to get today's weekday
    const todayDay = getGhanaWeekdayNumber(); // 1=Monday, 7=Sunday
    const dayName = getDayName(todayDay === 7 ? 0 : todayDay); // Convert to 0-6 format for day name

    Logger.log(`📅 Scanning timetable for day ${todayDay} (${dayName})`);

    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const day = parseInt(row[0], 10); // Timetable Day column (A)
      const startTime = formatTimeForCalculation(row[1]);
      const endTime = formatTimeForCalculation(row[2]);
      const teacher = row[3] ? row[3].toString().trim() : '';
      const subject = row[4] ? row[4].toString().trim() : '';
      const className = row[5] ? row[5].toString().trim() : '';

      if (!day || !startTime || !endTime || !teacher || !subject || !className) {
        continue;
      }

      // Match today's day with timetable day
      if (day === todayDay) {
        todaysLessons.push({
          teacher: teacher,
          subject: subject,
          className: className,
          startTime: startTime,
          endTime: endTime,
          period: `${startTime}-${endTime}`,
          day: dayName
        });
        
        Logger.log(`✅ SCHEDULED: ${teacher} - ${subject} - ${className} (${startTime}-${endTime})`);
      }
    }

    Logger.log(`📊 Total lessons scheduled for ${dayName}: ${todaysLessons.length}`);
    return todaysLessons;
    
  } catch (error) {
    Logger.log('ERROR in getTodaysLessons: ' + error.toString());
    return [];
  }
}


// Updated function to mark teacher as absent (daily version)
function markTeacherAbsentDaily(lesson, date, time) {
  try {
    // Final immediate re-check (race condition protection)
    const latestCheckDate = getGhanaDate();
    const foundNow = checkIfAttendanceSubmitted(lesson, latestCheckDate);
    
    if (foundNow) {
      Logger.log('Attendance was submitted in the final re-check — will NOT mark absent.');
      return;
    }

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let absenceSheet = ss.getSheetByName('Absences');
    if (!absenceSheet) {
      absenceSheet = ss.insertSheet('Absences');
      absenceSheet.getRange('A1:I1').setValues([[
        'Date', 'Time', 'Teacher', 'Subject', 'Class',
        'Scheduled Period', 'Status', 'Auto Generated', 'Reason'
      ]]);
    }

    // Standardize the date and time values we will store
    const tz = 'Africa/Accra';
    const now = new Date();
    const standardizedDate = Utilities.formatDate(now, tz, 'yyyy-MM-dd');
    const standardizedTime = Utilities.formatDate(now, tz, 'HH:mm');

    const existingData = absenceSheet.getDataRange().getValues();
    const isAlreadyRecorded = existingData.some(row =>
      // Check if absence already recorded for this teacher, class, subject and date
      row[0] === standardizedDate && 
      row[2] === lesson.teacher && 
      row[4] === lesson.className &&
      row[3] === lesson.subject
    );

    if (!isAlreadyRecorded) {
      absenceSheet.appendRow([
        standardizedDate,
        standardizedTime,
        lesson.teacher,
        lesson.subject,
        lesson.className,
        `${lesson.startTime}-${lesson.endTime} (${lesson.day})`,
        'Absent',
        'Yes',
        'Daily automated check'
      ]);

      sendAbsenceNotification(lesson, standardizedDate);
      Logger.log(`✅ Marked absent: ${lesson.teacher} - ${lesson.subject} - ${lesson.className} on ${lesson.day}`);
    } else {
      Logger.log('ℹ️ Absence already recorded for this teacher/class/subject/date; skipping duplicate.');
    }
  } catch (error) {
    Logger.log('Error in markTeacherAbsentDaily: ' + error.message);
  }
}

    const currentTimeObj = new Date(`1970-01-01T${currentTime}:00`);

    Logger.log(`🔍 Scanning timetable for day ${todayDay} at ${currentTime}`);

    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const day = parseInt(row[0], 10); // Timetable Day column (A)
      const startTime = formatTimeForCalculation(row[1]);
      const endTime = formatTimeForCalculation(row[2]);
      const teacher = row[3] ? row[3].toString().trim() : '';
      const subject = row[4] ? row[4].toString().trim() : '';
      const className = row[5] ? row[5].toString().trim() : '';

      if (!day || !startTime || !endTime || !teacher || !subject || !className) {
        continue;
      }

      // If timetable uses Monday=1 and getGhanaWeekdayNumber uses 1..7, this matches.
      if (day === todayDay) {
        const endTimeObj = new Date(`1970-01-01T${endTime}:00`);
        const timeDiff = (currentTimeObj - endTimeObj) / (1000 * 60); // minutes since lesson ended

        Logger.log(`Checking: ${teacher} - ${className} (${startTime}-${endTime})`);
        Logger.log(`  Ended ${timeDiff.toFixed(1)} minutes ago, Grace period: ${gracePeriodMinutes}min`);

        if (timeDiff > 0 && timeDiff <= gracePeriodMinutes) {
          recentLessons.push({
            teacher: teacher,
            subject: subject,
            className: className,
            startTime: startTime,
            endTime: endTime
          });
          Logger.log(`✅ ADD: ${teacher} - ${subject} - ${className}`);
        }
      }
    }

// Helper function to format time consistently for calculations
function formatTimeForCalculation(timeValue) {
  if (!timeValue) return '00:00';
  
  try {
    // If it's a Date object from Google Sheets
    if (timeValue instanceof Date) {
      const hours = timeValue.getHours().toString().padStart(2, '0');
      const minutes = timeValue.getMinutes().toString().padStart(2, '0');
      return `${hours}:${minutes}`;
    }
    
    // If it's already a string
    if (typeof timeValue === 'string') {
      // Handle "HH:MM" or "HH:MM:SS" format
      const timeMatch = timeValue.match(/(\d{1,2}):(\d{2})/);
      if (timeMatch) {
        const hours = timeMatch[1].padStart(2, '0');
        const minutes = timeMatch[2].padStart(2, '0');
        return `${hours}:${minutes}`;
      }
    }
    
    // If it's a number (Excel serial time)
    if (typeof timeValue === 'number') {
      const totalSeconds = Math.round(timeValue * 24 * 60 * 60);
      const hours = Math.floor(totalSeconds / 3600).toString().padStart(2, '0');
      const minutes = Math.floor((totalSeconds % 3600) / 60).toString().padStart(2, '0');
      return `${hours}:${minutes}`;
    }
    
    return '00:00';
  } catch (error) {
    Logger.log('Error formatting time: ' + error);
    return '00:00';
  }
}

// Get Ghana weekday number (1=Monday, 7=Sunday)
function getGhanaWeekdayNumber() {
  try {
    const timeZone = 'Africa/Accra';
    const now = new Date();
    const ghanaTime = Utilities.formatDate(now, timeZone, 'yyyy-MM-dd\'T\'HH:mm:ss');
    const ghanaDate = new Date(ghanaTime);
    
    // Convert to Ghana weekday (0=Sunday, 1=Monday, ..., 6=Saturday)
    let ghanaWeekday = ghanaDate.getDay();
    
    // Convert to match timetable format (1=Monday, 2=Tuesday, ..., 7=Sunday)
    if (ghanaWeekday === 0) {
      return 7; // Sunday
    } else {
      return ghanaWeekday; // Monday=1, Tuesday=2, ..., Saturday=6
    }
    
  } catch (error) {
    Logger.log('Error getting Ghana weekday: ' + error.message);
    // Fallback to local time
    const now = new Date();
    let weekday = now.getDay();
    return weekday === 0 ? 7 : weekday;
  }
}

// Check if attendance was submitted for a specific lesson
function checkIfAttendanceSubmitted(lesson, currentDate) {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    if (!sheet || sheet.getLastRow() < 2) {
      return false;
    }

    const data = sheet.getDataRange().getValues();
    // Column mapping (0-based index) per your description:
    // 0: DATETIME, 1: ACADEMIC YEAR, 2: SEMESTER, 3: NAME, 4: SUBJECT, 5: CLASS, ...
    const teacherName = lesson.teacher.trim().toLowerCase();
    const targetSubject = lesson.subject.trim().toLowerCase();
    const targetClass = lesson.className.trim().toLowerCase();

    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      if (!row[0]) continue;

      // parse row[0] into Ghana date string
      let recordDate;
      try {
        // If stored as Date object, use directly. If string, still parse.
        const recordDateObj = (row[0] instanceof Date) ? row[0] : new Date(row[0]);
        recordDate = Utilities.formatDate(recordDateObj, 'Africa/Accra', 'yyyy-MM-dd');
      } catch (e) {
        // skip malformed
        continue;
      }

      const recordTeacher = row[3] ? row[3].toString().trim().toLowerCase() : '';
      const recordSubject = row[4] ? row[4].toString().trim().toLowerCase() : '';
      const recordClassesRaw = row[5] ? row[5].toString().trim().toLowerCase() : '';

      // Normalize classes: split on comma and trim each
      const recordClasses = recordClassesRaw.split(',').map(c => c.trim()).filter(Boolean);

      if (recordDate === currentDate &&
          recordTeacher === teacherName &&
          recordSubject === targetSubject) {

        // If any class in recordClasses matches the targetClass as whole or as substring.
        for (let rc of recordClasses) {
          // exact match or record has multiple classes like "2A, 2B" so includes should work
          if (rc === targetClass || rc.indexOf(targetClass) !== -1 || targetClass.indexOf(rc) !== -1) {
            Logger.log(`✓ Attendance FOUND: ${recordTeacher} - ${recordSubject} - matched class ${rc}`);
            Logger.log(`  Record shows classes: ${recordClassesRaw}`);
            return true;
          }
        }
      }
    }

    Logger.log(`✗ Attendance NOT FOUND: ${teacherName} - ${targetSubject} - ${targetClass}`);
    return false;
  } catch (error) {
    Logger.log('ERROR in checkIfAttendanceSubmitted: ' + error.toString());
    return false;
  }
}

// Check for potential absences when teacher opens form

function checkPotentialAbsences(teacherName) {
  try {
    const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
    if (!absenceSheet) return [];
    
    const data = absenceSheet.getDataRange().getValues();
    Logger.log(`Total rows in Absences sheet: ${data.length}`);
    
    const absentLessons = [];
    
    // Today is just for reference
    const today = new Date();
    const todayStr = Utilities.formatDate(today, 'Africa/Accra', 'yyyy-MM-dd');
    Logger.log(`Today is: ${todayStr}`);
    
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      
      // Using your column structure:
      const absenceDate = row[0];      // A - Date
      const time = row[1];             // B - Time
      const teacher = row[2];          // C - Teacher
      const subject = row[3];          // D - Subject
      const className = row[4];        // E - Class
      const status = row[6];           // G - Status
      const autoGenerated = row[7];    // H - Auto Generated
      const reason = row[8];           // I - Reason
      
      // Skip if not this teacher
      if (teacher !== teacherName) {
        continue;
      }
      
      // Skip if not auto-generated
      if (autoGenerated !== 'Yes') {
        continue;
      }
      
      // Skip if already marked as resolved
      if (status && 
          (status.includes('Remedial') || 
           status.includes('Made Up') || 
           status.includes('Cleared') || 
           status.includes('Verified'))) {
        continue;
      }
      
      // Format the date
      let absenceDateStr;
      try {
        if (absenceDate instanceof Date) {
          absenceDateStr = Utilities.formatDate(absenceDate, 'Africa/Accra', 'yyyy-MM-dd');
        } else if (typeof absenceDate === 'string') {
          const dateObj = new Date(absenceDate);
          absenceDateStr = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
        } else {
          continue;
        }
      } catch (e) {
        continue;
      }
      
      // Check if this absence is within the last 30 days (for relevance)
      const absenceDateObj = new Date(absenceDateStr);
      const thirtyDaysAgo = new Date();
      thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);
      
      if (absenceDateObj < thirtyDaysAgo) {
        continue; // Skip absences older than 30 days
      }
      
      // Add to list - this absence needs a remedial lesson
      absentLessons.push({
        date: absenceDateStr,
        subject: subject || 'Unknown',
        className: className || 'Unknown',
        time: time || 'Unknown',
        reason: reason || 'No reason provided',
        status: status || 'Absent',
        autoGenerated: autoGenerated
      });
      
      Logger.log(`Found absence for ${teacherName}: ${absenceDateStr} - ${subject} - ${className}`);
    }
    
    Logger.log(`Total absences found for ${teacherName}: ${absentLessons.length}`);
    
    // Sort by date (most recent first)
    absentLessons.sort((a, b) => new Date(b.date) - new Date(a.date));
    
    return absentLessons;
    
  } catch (error) {
    Logger.log('Error checking potential absences: ' + error.message);
    return [];
  }
}

// Function to reverse/remove absence records
function reverseAbsence(absenceData) {
  try {
    Logger.log('=== STARTING ABSENCE REVERSAL ===');
    Logger.log('Absence data received:', JSON.stringify(absenceData));
    
    const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
    if (!absenceSheet) {
      throw new Error('Absences sheet not found');
    }
    
    const lastRow = absenceSheet.getLastRow();
    if (lastRow < 2) {
      throw new Error('No absence records found');
    }
    
    // Get all absence data
    const data = absenceSheet.getDataRange().getValues();
    const headers = data[0];
    
    Logger.log(`Scanning ${data.length - 1} absence records...`);
    
    let recordsReversed = 0;
    const reversedRecords = [];
    
    // Find and remove matching absence records
    for (let i = data.length - 1; i >= 1; i--) { // Start from bottom for deletion
      const row = data[i];
      const recordDate = row[0];
      const recordTeacher = row[2];
      const recordSubject = row[3];
      const recordClass = row[4];
      const recordTime = row[5];
      const autoGenerated = row[7];
      
      // Convert date to comparable format
      let recordDateFormatted = '';
      if (recordDate instanceof Date) {
        recordDateFormatted = Utilities.formatDate(recordDate, 'Africa/Accra', 'yyyy-MM-dd');
      } else if (typeof recordDate === 'string') {
        try {
          const dateObj = new Date(recordDate);
          recordDateFormatted = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
        } catch (e) {
          recordDateFormatted = recordDate;
        }
      }
      
      const targetDateFormatted = absenceData.date;
      
      // Check if this record matches the criteria for reversal
      const isDateMatch = recordDateFormatted === targetDateFormatted;
      const isTeacherMatch = recordTeacher === absenceData.teacherName;
      const isAutoGenerated = autoGenerated === 'Yes';
      
      // For subject/class matching, be flexible since the teacher might have multiple classes
      const isSubjectMatch = !absenceData.subject || recordSubject === absenceData.subject;
      const isClassMatch = !absenceData.className || recordClass === absenceData.className;
      
      Logger.log(`Checking record ${i}: ${recordTeacher} - ${recordDateFormatted}`);
      Logger.log(`  Date match: ${isDateMatch}, Teacher match: ${isTeacherMatch}, Auto-generated: ${isAutoGenerated}`);
      
      if (isDateMatch && isTeacherMatch && isAutoGenerated && isSubjectMatch && isClassMatch) {
        // Delete the row
        absenceSheet.deleteRow(i + 1);
        recordsReversed++;
        
        reversedRecords.push({
          date: recordDateFormatted,
          teacher: recordTeacher,
          subject: recordSubject,
          className: recordClass,
          time: recordTime
        });
        
        Logger.log(`✅ REVERSED ABSENCE: Row ${i + 1} - ${recordTeacher} - ${recordSubject}`);
      }
    }
    
    Logger.log(`=== ABSENCE REVERSAL COMPLETED: ${recordsReversed} records reversed ===`);
    
    return {
      success: true,
      recordsReversed: recordsReversed,
      reversedRecords: reversedRecords,
      message: `Successfully reversed ${recordsReversed} absence record(s) for ${absenceData.teacherName}`
    };
    
  } catch (error) {
    Logger.log('❌ ERROR in reverseAbsence: ' + error.toString());
    Logger.log('Stack: ' + error.stack);
    return {
      success: false,
      recordsReversed: 0,
      reversedRecords: [],
      message: 'Failed to reverse absence: ' + error.message
    };
  }
}

    // Get today's attendance count
    const formDataSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    const formData = formDataSheet.getDataRange().getValues();
    const todayAttendance = formData.filter((row, index) => {
      if (index === 0) return false; // Skip header
      const recordDate = Utilities.formatDate(new Date(row[0]), Session.getScriptTimeZone(), 'yyyy-MM-dd');
      return recordDate === todayStr;
    }).length;
    
    // Get today's absences
    const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
    let todayAbsences = 0;
    if (absenceSheet) {
      const absenceData = absenceSheet.getDataRange().getValues();
      todayAbsences = absenceData.filter((row, index) => {
        if (index === 0) return false;
        const absenceDate = Utilities.formatDate(new Date(row[0]), Session.getScriptTimeZone(), 'yyyy-MM-dd');
        return absenceDate === todayStr;
      }).length;
    }
    
    // Get total teachers
    const teachers = getTeacherNames();
    
    // Get this week's records
    const weekStart = new Date(today);
    weekStart.setDate(today.getDate() - today.getDay());
    const weekRecords = formData.filter((row, index) => {
      if (index === 0) return false;
      const recordDate = new Date(row[0]);
      return recordDate >= weekStart;
    }).length;
    
    return {
      todayAttendance: todayAttendance,
      todayAbsences: todayAbsences,
      totalTeachers: teachers.length,
      weekAttendance: weekRecords
    };
  } catch (error) {
    Logger.log('Error in getAdminStats: ' + error.message);
    return {
      todayAttendance: 0,
      todayAbsences: 0,
      totalTeachers: 0,
      weekAttendance: 0
    };
  }
}

// Get recent attendance records
function getRecentAttendance(limit = 10) {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    const data = sheet.getDataRange().getValues();

    if (data.length <= 1) return []; // No data except header

    // Remove header row
    const records = data.slice(1);
    const recentRecords = records.slice(-limit).reverse();

    const formattedRecords = recentRecords.map((row, index) => {
      const record = {};

      try {
        // Safely format Date (A)
        if (row[0]) {
          const parsedDate = new Date(row[0]);
          record.date = isNaN(parsedDate)
            ? String(row[0])
            : Utilities.formatDate(parsedDate, Session.getScriptTimeZone(), 'yyyy-MM-dd');
        } else {
          record.date = 'N/A';
        }

        // Safely format Time (B)
        if (row[1]) {
          const parsedTime = new Date(row[1]);
          record.time = isNaN(parsedTime)
            ? String(row[1])
            : Utilities.formatDate(parsedTime, Session.getScriptTimeZone(), 'HH:mm');
        } else {
          record.time = 'N/A';
        }

        record.teacher = row[4] ? String(row[4]).trim() : 'N/A';     // NAME (E)
        record.subject = row[5] ? String(row[5]).trim() : 'N/A';     // SUBJECT (F)
        record.classNames = row[6] ? String(row[6]).trim() : 'N/A';  // CLASS (G)
        record.periods = row[7] ? String(row[7]).trim() : 'N/A';     // PERIODS (H)
        record.topic = row[8] ? String(row[8]).trim() : 'N/A';       // TOPIC (I)
        record.location = row[12] ? String(row[12]).trim() : 'N/A';  // LOCATION (M)

      } catch (err) {
        Logger.log(`Error processing row ${index}: ${err.message}`);
      }

      return record;
    });

    Logger.log(`Returning ${formattedRecords.length} recent records`);
    return formattedRecords;

  } catch (error) {
    Logger.log('Error in getRecentAttendance: ' + error.message);
    Logger.log('Stack trace: ' + error.stack);
    return [];
  }
}

// CORRECTED VERSION: Get classroom status from timetable data
function getClassroomStatus() {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const timetableSheet = ss.getSheetByName('Timetable');
    const formDataSheet = ss.getSheetByName('FormData');
    
    if (!timetableSheet || !formDataSheet) {
      Logger.log('ERROR: Timetable or FormData sheet not found');
      return [];
    }
    
    // Get current Ghana time and date
    const now = new Date();
    const currentDate = Utilities.formatDate(now, 'Africa/Accra', 'yyyy-MM-dd');
    const currentTime = Utilities.formatDate(now, 'Africa/Accra', 'HH:mm');
    const currentDay = getGhanaWeekdayNumber();
    
    Logger.log(`=== CLASSROOM STATUS CHECK ===`);
    Logger.log(`Date: ${currentDate}, Time: ${currentTime}, Day: ${currentDay}`);
    
    // Load timetable data
    const timetableData = timetableSheet.getDataRange().getValues();
    const allClassrooms = new Set();
    
    // Extract all individual classrooms from the timetable
    for (let i = 1; i < timetableData.length; i++) {
      const rawClass = timetableData[i][5]; // Column F - Class
      if (!rawClass) continue;
      
      // Split classes by comma and trim each one
      const classList = rawClass.toString().split(',').map(c => c.trim());
      classList.forEach(c => {
        if (c !== '') allClassrooms.add(c);
      });
    }
    
    Logger.log(`Found ${allClassrooms.size} classrooms: ${Array.from(allClassrooms).join(', ')}`);
    
    // Load today's attendance records
    const attendanceData = formDataSheet.getDataRange().getValues();
    const todaysAttendance = [];
    
    for (let i = 1; i < attendanceData.length; i++) {
      const row = attendanceData[i];
      const recordDate = row[0];
      if (!recordDate) continue;
      
      let recordDateStr;
      try {
        if (recordDate instanceof Date) {
          recordDateStr = Utilities.formatDate(recordDate, 'Africa/Accra', 'yyyy-MM-dd');
        } else {
          recordDateStr = Utilities.formatDate(new Date(recordDate), 'Africa/Accra', 'yyyy-MM-dd');
        }
      } catch (e) {
        continue;
      }
      
      if (recordDateStr === currentDate) {
        todaysAttendance.push({
          teacher: row[4] ? row[4].toString().trim() : '', // Column E - NAME
          subject: row[5] ? row[5].toString().trim() : '', // Column F - SUBJECT
          classes: row[6] ? row[6].toString().split(',').map(c => c.trim()) : [], // Column G - CLASS
          time: row[1] ? row[1].toString().trim() : '' // Column B - TIME
        });
      }
    }
    
    Logger.log(`Found ${todaysAttendance.length} attendance records for today`);
    
    // Find classes currently in session based on timetable
    const currentPeriodClasses = [];
    
    for (let i = 1; i < timetableData.length; i++) {
      const row = timetableData[i];
      const day = parseInt(row[0], 10); // Column A - Day
      const startTime = formatTimeForCalculation(row[1]); // Column B - StartTime
      const endTime = formatTimeForCalculation(row[2]); // Column C - EndTime
      const teacher = row[3] ? row[3].toString().trim() : ''; // Column D - Teacher
      const subject = row[4] ? row[4].toString().trim() : ''; // Column E - Subject
      const rawClass = row[5] ? row[5].toString().trim() : ''; // Column F - Class
      
      if (!teacher || !rawClass || !startTime || !endTime) continue;

      // Check if this lesson is happening right now
      if (day === currentDay && isTimeInRange(currentTime, startTime, endTime)) {
        const classList = rawClass.split(',').map(c => c.trim());
        
        classList.forEach(className => {
          if (className) {
            currentPeriodClasses.push({
              className: className,
              teacher: teacher,
              subject: subject,
              startTime: startTime,
              endTime: endTime,
              period: `${startTime}-${endTime}`,
              rawClasses: rawClass
            });
          }
        });

        Logger.log(`Current period: ${rawClass} - ${teacher} - ${subject} (${startTime}-${endTime})`);
      }
    }
    
    Logger.log(`Found ${currentPeriodClasses.length} classes in current period`);
    
    // Build classroom status list
    const classroomStatus = [];
    
    allClassrooms.forEach(classroomName => {
      // Check if this classroom has a scheduled class this period
      const scheduledLesson = currentPeriodClasses.find(lesson => 
        lesson.className === classroomName
      );
      
      if (scheduledLesson) {
        // Check if attendance was submitted for this class
        const attendanceRecord = todaysAttendance.find(record => {
          return record.teacher === scheduledLesson.teacher && 
                 record.classes.includes(classroomName) &&
                 record.subject === scheduledLesson.subject;
        });
        
        const isOccupied = !!attendanceRecord;
        
        classroomStatus.push({
          name: classroomName,
          status: isOccupied ? 'occupied' : 'vacant',
          teacher: scheduledLesson.teacher,
          scheduledTeacher: scheduledLesson.teacher,
          subject: scheduledLesson.subject,
          time: scheduledLesson.period,
          isCurrentPeriod: true,
          hasAttendance: isOccupied,
          attendanceTime: isOccupied ? formatAttendanceTime(attendanceRecord.time) : '',
          rawClasses: scheduledLesson.rawClasses
        });
        
        Logger.log(`Classroom ${classroomName}: ${isOccupied ? 'OCCUPIED' : 'VACANT'} - ${scheduledLesson.teacher}`);
        
      } else {
        // No class scheduled now
        classroomStatus.push({
          name: classroomName,
          status: 'vacant',
          teacher: '',
          scheduledTeacher: '',
          subject: '',
          time: '',
          isCurrentPeriod: false,
          hasAttendance: false,
          attendanceTime: '',
          rawClasses: ''
        });
        
        Logger.log(`Classroom ${classroomName}: NO SCHEDULE`);
      }
    });
    
    // Sort: occupied first → then by name
    classroomStatus.sort((a, b) => {
      if (a.status === b.status) return a.name.localeCompare(b.name);
      return a.status === 'occupied' ? -1 : 1;
    });
    
    Logger.log(`FINAL: ${classroomStatus.length} classrooms processed.`);
    Logger.log(`Occupied: ${classroomStatus.filter(c => c.status === 'occupied').length}`);
    Logger.log(`Vacant (scheduled): ${classroomStatus.filter(c => c.status === 'vacant' && c.isCurrentPeriod).length}`);
    Logger.log(`Vacant (no schedule): ${classroomStatus.filter(c => c.status === 'vacant' && !c.isCurrentPeriod).length}`);
    
    return classroomStatus;
    
  } catch (error) {
    Logger.log('❌ ERROR in getClassroomStatus: ' + error.toString());
    Logger.log('Stack: ' + error.stack);
    return [];
  }
}

function formatAttendanceTime(timeValue) {
  try {
    if (!timeValue) return 'Not recorded';
    
    // If it's already a string in HH:mm:ss format
    if (typeof timeValue === 'string') {
      // Extract just the time part (HH:mm)
      const timeMatch = timeValue.match(/(\d{1,2}:\d{2})/);
      if (timeMatch) {
        return timeMatch[1]; // Returns just "15:15" instead of full date string
      }
      return timeValue;
    }
    
    // If it's a Date object
    if (timeValue instanceof Date) {
      return Utilities.formatDate(timeValue, 'Africa/Accra', 'HH:mm');
    }
    
    return 'Time recorded';
  } catch (error) {
    Logger.log('Error formatting attendance time: ' + error);
    return 'Time recorded';
  }
}

// Improved time range checking
function isTimeInRange(currentTime, startTime, endTime) {
  try {
    // Format times consistently
    const formatTime = (timeStr) => {
      if (!timeStr) return '00:00';
      // Remove seconds if present
      const parts = timeStr.split(':');
      if (parts.length >= 2) {
        return `${parts[0].padStart(2, '0')}:${parts[1].padStart(2, '0')}`;
      }
      return '00:00';
    };

    const current = formatTime(currentTime);
    const start = formatTime(startTime);
    const end = formatTime(endTime);

    Logger.log(`Time check: Current=${current}, Start=${start}, End=${end}`);
    
    // Simple string comparison for HH:MM format
    return current >= start && current <= end;
  } catch (e) {
    Logger.log(`Time range error: ${e}`);
    return false;
  }
}

// Manual attendance submission for admin
function submitManualAttendance(data) {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    
    // Get current date and time
    const timeZone = SpreadsheetApp.getActiveSpreadsheet().getSpreadsheetTimeZone();
    const now = new Date();
    const date = Utilities.formatDate(now, timeZone, 'yyyy-MM-dd');
    const time = Utilities.formatDate(now, timeZone, 'HH:mm:ss');
    
    // Calculate week number
    const weekNumber = getWeekNumber(now);

    // Validate required fields
    if (!data.name || !data.subject || !data.classNames) {
      throw new Error('Teacher name, subject, and class are required');
    }

    // Get academic year and semester if not provided
    let academicYear = data.academicYear || getCurrentAcademicYear();
    let semester = data.semester || getCurrentSemester();
    
    if (!academicYear) {
      throw new Error('Academic year is required');
    }
    
    if (!semester) {
      throw new Error('Semester is required');
    }

    const name = data.name;
    const subject = data.subject;
    const classNames = data.classNames;
    const periods = data.periods || 1;
    const topic = data.topic || 'Manual Entry by Admin';
    const teachingLocation = data.teachingLocation || 'Manual Entry';

    // Check for duplicates (same teacher, subject, class, date)
    const lastRow = sheet.getLastRow();
    if (lastRow > 1) {
      const dataRange = sheet.getRange(2, 1, lastRow - 1, 7);
      const values = dataRange.getValues();

      for (let i = 0; i < values.length; i++) {
        const recordedDate = Utilities.formatDate(new Date(values[i][0]), timeZone, 'yyyy-MM-dd');
        const recordedName = values[i][4];
        const recordedSubject = values[i][5];
        const recordedClasses = values[i][6];

        if (recordedDate === date && recordedName === name && recordedSubject === subject) {
          const recordedClassArray = recordedClasses.split(', ').map(c => c.trim());
          const currentClassArray = classNames.split(', ').map(c => c.trim());

          if (currentClassArray.some(cls => recordedClassArray.includes(cls))) {
            throw new Error(
              `Attendance already recorded for ${name} in overlapping class(es) (${recordedClasses}) for this subject today.`
            );
          }
        }
      }
    }

    // Create a placeholder image for manual entries
    const placeholderImageUrl = createPlaceholderImage(name, subject, classNames, date);
    
    // Append the data to the Google Sheet
    sheet.appendRow([
      date, time, academicYear, semester, name, 
      subject, classNames, periods, topic, 
      'Manual Entry - No GPS', placeholderImageUrl, weekNumber, teachingLocation, 'Manual entry by admin'
    ]);

    // Set the URL as a chip in the last row
    const newRow = sheet.getLastRow();
    const urlCell = sheet.getRange(newRow, 11);
    urlCell.setFormula('=HYPERLINK("' + placeholderImageUrl + '", "View Entry")');

    // If this was to reverse an absence, remove the absence record
    if (data.reverseAbsence) {
      try {
        const absenceData = {
          teacherName: name,
          date: date
        };
        reverseAbsence(absenceData);
      } catch (absenceError) {
        Logger.log('Note: Could not reverse absence record: ' + absenceError.message);
      }
    }

    return {
      success: true,
      message: `Manual attendance recorded successfully for ${name}`,
      record: {
        date: date,
        time: time,
        teacher: name,
        subject: subject,
        classNames: classNames,
        periods: periods
      }
    };
    
  } catch (error) {
    Logger.log('Error in submitManualAttendance: ' + error.message);
    throw new Error('Failed to submit manual attendance: ' + error.message);
  }
}

// Get Teacher Attendance Summary with cumulative data since LessonsStartDate
function getTeacherAttendanceSummary() {
  try {
    const teachers = getTeacherNames();
    const summary = [];
    
    // Get current academic year and semester
    const academicYearData = getAcademicYearsAndSemesters();
    const currentYear = academicYearData.currentYear;
    const currentSemester = academicYearData.currentSemester;
    
    if (!currentYear || !currentSemester) {
      throw new Error('Current academic year or semester not found');
    }
    
    // Get lessons start date for current semester
    const lessonsStartDate = getLessonsStartDate(currentYear, currentSemester);
    if (!lessonsStartDate) {
      throw new Error('Lessons start date not found for current semester');
    }
    
    const today = new Date();
    
    teachers.forEach(teacher => {
      const teacherData = calculateCumulativeTeacherAttendance(
        teacher, 
        currentYear, 
        currentSemester, 
        lessonsStartDate, 
        today
      );
      summary.push(teacherData);
    });
    
    // Sort by attendance percentage (descending)
    summary.sort((a, b) => b.attendancePercentage - a.attendancePercentage);
    
    return summary;
  } catch (error) {
    Logger.log('Error in getTeacherAttendanceSummary: ' + error.message);
    return [];
  }
}

// Get lessons start date for specific academic year and semester

    const academicYearSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('AcademicYear');
    const data = academicYearSheet.getDataRange().getValues();
    
    for (let i = 1; i < data.length; i++) {

      const sem = convertSemesterToNumber(row[2]);

// Calculate cumulative attendance since lessons start date
function calculateCumulativeTeacherAttendance(teacherName, academicYear, semester, startDate, endDate) {
  try {
    Logger.log(`\n=== CALCULATING ATTENDANCE FOR: ${teacherName} ===`);
    Logger.log(`Date Range: ${startDate} to ${endDate}`);
    Logger.log(`Academic Year: ${academicYear}, Semester: ${semester}`);
    
    // Get all scheduled periods since lessons start date
    const scheduledPeriods = getCumulativeScheduledPeriods(teacherName, startDate, endDate);
    
    // Count attended periods within the date range for the specific academic year/semester
    const attendedPeriods = countCumulativeAttendedPeriods(teacherName, academicYear, semester, startDate, endDate);
    
    // Count absent periods from Absences sheet
    const absentPeriods = countCumulativeAbsentPeriods(teacherName, academicYear, semester, startDate, endDate);
    
    // Calculate unattended periods (scheduled but no record)
    const unattendedPeriods = Math.max(0, scheduledPeriods - (attendedPeriods + absentPeriods));
    
    // Calculate percentages
    const attendancePercentage = scheduledPeriods > 0 ? 
      Math.round((attendedPeriods / scheduledPeriods) * 100) : 0;
    
    const accountabilityPercentage = scheduledPeriods > 0 ?
      Math.round(((attendedPeriods + absentPeriods) / scheduledPeriods) * 100) : 0;
    
    Logger.log(`Results for ${teacherName}:`);
    Logger.log(`  Scheduled: ${scheduledPeriods}`);
    Logger.log(`  Attended: ${attendedPeriods}`);
    Logger.log(`  Absent: ${absentPeriods}`);
    Logger.log(`  Unattended: ${unattendedPeriods}`);
    Logger.log(`  Attendance %: ${attendancePercentage}`);
    
    return {
      name: teacherName,
      scheduledPeriods: scheduledPeriods,
      attendedPeriods: attendedPeriods,
      absentPeriods: absentPeriods,
      unattendedPeriods: unattendedPeriods,
      attendancePercentage: attendancePercentage,
      accountabilityPercentage: accountabilityPercentage,
      dateRange: `${formatDateDisplay(startDate)} to ${formatDateDisplay(endDate)}`,
      academicYear: academicYear,
      semester: `Semester ${semester}`
    };
    
  } catch (error) {
    Logger.log('Error calculating cumulative attendance for ' + teacherName + ': ' + error.message);
    return getFallbackData(teacherName);
  }
}


// Get cumulative scheduled periods since start date
function getCumulativeScheduledPeriods(teacherName, startDate, endDate) {
  try {
    const timetableSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Timetable');
    if (!timetableSheet) return 0;
    
    const data = timetableSheet.getDataRange().getValues();
    let totalPeriods = 0;
    
    // Get teacher's weekly schedule pattern
    const weeklySchedule = getTeacherWeeklySchedule(teacherName);
    
    if (weeklySchedule.length === 0) {
      Logger.log(`No weekly schedule found for ${teacherName}`);
      return 0;
    }
    
    Logger.log(`Weekly schedule for ${teacherName}:`, JSON.stringify(weeklySchedule));
    
    // Calculate all school days between start and end dates
    const schoolDays = getAllSchoolDays(startDate, endDate);
    Logger.log(`School days in range: ${schoolDays.length}`);
    
    // For each school day, check if teacher has scheduled classes
    schoolDays.forEach(schoolDay => {
      const dayOfWeek = schoolDay.getDay(); // 0=Sunday, 1=Monday, etc.
      
      // Convert to timetable format (1=Monday, 7=Sunday)
      const timetableDay = dayOfWeek === 0 ? 7 : dayOfWeek;
      
      // Check if teacher has schedule for this day of week
      const daySchedule = weeklySchedule.filter(schedule => schedule.day === timetableDay);
      
      if (daySchedule.length > 0) {
        // Count periods for this day
        const dayPeriods = daySchedule.reduce((total, schedule) => total + (schedule.periods || 0), 0);
        
        totalPeriods += dayPeriods;

// Get teacher's weekly schedule pattern (days and periods)
function getTeacherWeeklySchedule(teacherName) {
  try {
    const timetableSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Timetable');
    const lastRow = timetableSheet.getLastRow();
    
    if (lastRow < 2) return [];
    
    const data = timetableSheet.getRange(2, 1, lastRow - 1, 6).getValues();
    
    const weeklySchedule = [];
    
    data.forEach(row => {
      const day = parseInt(row[0]);
      const teacher = row[3] ? row[3].toString().trim() : '';
      const startTime = formatTime(row[1]);
      const endTime = formatTime(row[2]);
      
      if (teacher === teacherName && day) {
        const periods = calculatePeriodsFromTime(startTime, endTime);
        
        // Check if we already have this day in schedule
        const existingDay = weeklySchedule.find(schedule => schedule.day === day);
        
        if (existingDay) {
          existingDay.periods += periods;
        } else {
          weeklySchedule.push({
            day: day,
            periods: periods
          });
        }
      }
    });
    
    return weeklySchedule;
  } catch (error) {
    Logger.log('Error getting teacher weekly schedule: ' + error.message);
    return [];
  }
}


// Get all school days (Monday-Friday) between two dates
function getAllSchoolDays(startDate, endDate) {
  const schoolDays = [];
  const currentDate = new Date(startDate);
  const end = new Date(endDate);
  
  // Ensure we're working with dates at start of day
  currentDate.setHours(0, 0, 0, 0);
  end.setHours(0, 0, 0, 0);
  
  while (currentDate <= end) {
    const dayOfWeek = currentDate.getDay();
    
    // Only count weekdays (Monday=1 to Friday=5)
    if (dayOfWeek >= 1 && dayOfWeek <= 5) {
      schoolDays.push(new Date(currentDate));
    }
    
    currentDate.setDate(currentDate.getDate() + 1);
  }
  
  return schoolDays;
}

function getCurrentWeekDatesFromAcademicYear() {
  try {
    const academicYearSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('AcademicYear');
    if (!academicYearSheet) {
      throw new Error('AcademicYear sheet not found');
    }
    
    const data = academicYearSheet.getDataRange().getValues();
    
    // Find the current semester's lessons start date (Column E)
    let lessonsStartDate = null;
    const today = new Date();
    
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const startDate = row[1]; // Column B - StartDate
      const endDate = row[2];   // Column C - EndDate
      const lessonsStart = row[4]; // Column E - LessonsStartDate
      
      if (startDate && endDate && lessonsStart) {
        const semesterStart = new Date(startDate);
        const semesterEnd = new Date(endDate);
        
        // Check if current date falls within this semester
        if (today >= semesterStart && today <= semesterEnd) {
          lessonsStartDate = new Date(lessonsStart);
          break;
        }
      }
    }
    
    // If no current semester found, use default calculation
    if (!lessonsStartDate) {
      Logger.log('No current semester found, using default week calculation');
      return getCurrentWeekDates();
    }
    
    // Calculate current week number based on lessons start date
    const millisecondsPerWeek = 7 * 24 * 60 * 60 * 1000;
    const weeksSinceStart = Math.floor((today - lessonsStartDate) / millisecondsPerWeek);
    const currentWeekStart = new Date(lessonsStartDate.getTime() + (weeksSinceStart * millisecondsPerWeek));
    
    // Generate dates for the current week (Monday to Friday)
    const weekDates = [];
    for (let i = 0; i < 5; i++) {
      const date = new Date(currentWeekStart);
      date.setDate(currentWeekStart.getDate() + i);
      weekDates.push(Utilities.formatDate(date, Session.getScriptTimeZone(), 'yyyy-MM-dd'));
    }
    
    Logger.log('Current week dates based on academic year: ' + weekDates.join(', '));
    return weekDates;
    
  } catch (error) {
    Logger.log('Error in getCurrentWeekDatesFromAcademicYear: ' + error.message);
    return getCurrentWeekDates(); // Fallback to default calculation
  }
}

function getScheduledPeriods(teacherName, dateRange) {

    for (let i = 1; i < data.length; i++) {

function calculateTeacherAttendance(teacherName, dateRange) {
  try {
    const formDataSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
    
    if (!formDataSheet || formDataSheet.getLastRow() < 2) {
      return {
        name: teacherName,
        totalPeriods: 0,
        presentPeriods: 0,
        absentPeriods: 0,
        attendancePercentage: 0
      };
    }
    
    const formData = formDataSheet.getDataRange().getValues();
    let presentPeriods = 0;
    
    // Count present periods for this teacher in date range
    for (let i = 1; i < formData.length; i++) {

      name: teacherName,

      absentPeriods: absentPeriods,

// Check if date is within range
function isDateInRange(dateObj, startDate, endDate) {
  try {
    const date = new Date(dateObj);
    const start = new Date(startDate);
    const end = new Date(endDate);
    
    // Set all dates to start of day for proper comparison
    date.setHours(0, 0, 0, 0);
    start.setHours(0, 0, 0, 0);
    end.setHours(0, 0, 0, 0);
    
    return date >= start && date <= end;
  } catch (error) {
    Logger.log('Error in isDateInRange: ' + error.message);
    return false;
  }
}

// Calculate periods from start and end time (1 hour = 1 period, ignore extra minutes)
function calculatePeriodsFromTime(startTime, endTime) {
  try {
    if (!startTime || !endTime) return 0;
    
    // Parse times (assuming format "HH:mm")
    const [startHours, startMinutes] = startTime.split(':').map(Number);
    const [endHours, endMinutes] = endTime.split(':').map(Number);
    
    // Calculate total minutes
    const startTotalMinutes = startHours * 60 + startMinutes;
    const endTotalMinutes = endHours * 60 + endMinutes;
    
    // Calculate duration in minutes
    let durationMinutes = endTotalMinutes - startTotalMinutes;
    
    // If duration is negative, assume it crosses midnight (unlikely for school schedule)
    if (durationMinutes < 0) {
      durationMinutes += 24 * 60; // Add 24 hours
    }
    
    // Calculate full hours (ignore extra minutes)
    const fullHours = Math.floor(durationMinutes / 60);
    
    Logger.log(`Time: ${startTime}-${endTime}, Duration: ${durationMinutes}min, Periods: ${fullHours}`);
    
    return fullHours;
    
  } catch (error) {
    Logger.log('Error calculating periods from time: ' + error.message);
    return 0;
  }
}

    Sunday: 0

  schedule.forEach(entry => {

// Check for schedule conflicts
function checkScheduleConflict(scheduleData) {
  try {
    const timetableSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Timetable');
    const lastRow = timetableSheet.getLastRow();
    
    if (lastRow < 2) return [];
    
    const data = timetableSheet.getRange(2, 1, lastRow - 1, 6).getValues();
    const conflicts = [];
    
    const newStart = new Date(`1970-01-01T${scheduleData.startTime}`);
    const newEnd = new Date(`1970-01-01T${scheduleData.endTime}`);
    
    data.forEach(row => {
      const day = parseInt(row[0]);
      const existingStart = new Date(`1970-01-01T${formatTime(row[1])}`);
      const existingEnd = new Date(`1970-01-01T${formatTime(row[2])}`);
      const teacher = row[3];
      const className = row[5];
      
      // Check same day
      if (day === parseInt(scheduleData.day)) {
        // Check teacher conflict
        if (teacher === scheduleData.teacher) {
          if ((newStart >= existingStart && newStart < existingEnd) || 
              (newEnd > existingStart && newEnd <= existingEnd) ||
              (newStart <= existingStart && newEnd >= existingEnd)) {
            const existingPeriods = calculatePeriodsFromTime(formatTime(row[1]), formatTime(row[2]));
            conflicts.push(`Teacher already has ${existingPeriods}-period class at ${formatTime(row[1])}-${formatTime(row[2])}`);
          }
        }
        
        // Check classroom conflict
        if (className === scheduleData.className) {
          if ((newStart >= existingStart && newStart < existingEnd) || 
              (newEnd > existingStart && newEnd <= existingEnd) ||
              (newStart <= existingStart && newEnd >= existingEnd)) {
            const existingPeriods = calculatePeriodsFromTime(formatTime(row[1]), formatTime(row[2]));
            conflicts.push(`Classroom ${className} already occupied for ${existingPeriods} period${existingPeriods !== 1 ? 's' : ''} at ${formatTime(row[1])}-${formatTime(row[2])}`);
          }
        }
      }
    });
    
    return conflicts;
    
  } catch (error) {
    Logger.log('Error checking schedule conflict: ' + error.message);
    return ['Error checking conflicts'];
  }
}

/**
 * Clears all false absence records that were auto-generated but have matching attendance
 * This fixes the issue caused by the time formatting bug that created false absences
 */
function clearAllFalseAbsences() {
  try {
    Logger.log('=== CLEARING ALL FALSE ABSENCES ===');
    
    const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
    const formDataSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    
    if (!absenceSheet) {
      throw new Error('Absences sheet not found');
    }
    
    const absenceData = absenceSheet.getDataRange().getValues();
    const formData = formDataSheet.getDataRange().getValues();
    
    let deletedCount = 0;
    let checkedCount = 0;
    const falseAbsences = [];
    
    Logger.log(`Scanning ${absenceData.length - 1} absence records...`);
    
    // Collect false absences first (delete from bottom to top)
    for (let i = absenceData.length - 1; i >= 1; i--) {
      const row = absenceData[i];
      const absenceDate = row[0];
      const teacher = row[2];
      const subject = row[3];
      const className = row[4];
      const autoGenerated = row[7];
      
      // Only check auto-generated records
      if (autoGenerated === 'Yes') {
        checkedCount++;
        
        // Convert absence date to comparable format
        let absenceDateFormatted = '';
        try {
          if (absenceDate instanceof Date) {
            absenceDateFormatted = Utilities.formatDate(absenceDate, 'Africa/Accra', 'yyyy-MM-dd');
          } else if (typeof absenceDate === 'string') {
            const dateObj = new Date(absenceDate);
            absenceDateFormatted = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
          }
        } catch (e) {
          continue;
        }
        
        // Check if attendance exists for this teacher, subject, and date
        const hasAttendance = checkAttendanceExists(formData, teacher, subject, className, absenceDateFormatted);
        
        if (hasAttendance) {
          falseAbsences.push({
            rowIndex: i + 1,
            teacher: teacher,
            subject: subject,
            className: className,
            date: absenceDateFormatted
          });
        }
      }
    }
    
    Logger.log(`Found ${falseAbsences.length} false absences out of ${checkedCount} auto-generated records`);
    
    // Delete the false absences
    falseAbsences.forEach(absence => {
      absenceSheet.deleteRow(absence.rowIndex);
      deletedCount++;
      Logger.log(`🗑️ Deleted false absence: ${absence.teacher} - ${absence.subject} - ${absence.className} (${absence.date})`);
    });
    
    const resultMessage = `✅ Successfully cleared ${deletedCount} false absence records`;
    Logger.log(resultMessage);
    
    // Generate detailed report
    generateFalseAbsenceReport(falseAbsences);
    
    return {
      success: true,
      deletedCount: deletedCount,
      checkedCount: checkedCount,
      message: resultMessage
    };
    
  } catch (error) {
    Logger.log('❌ ERROR in clearAllFalseAbsences: ' + error.toString());
    return {
      success: false,
      deletedCount: 0,
      checkedCount: 0,
      message: 'Failed to clear false absences: ' + error.message
    };
  }
}

/**
 * Checks if attendance exists for a given teacher, subject, class and date
 */
function checkAttendanceExists(formData, teacher, subject, className, date) {
  try {
    const teacherLower = teacher.toLowerCase().trim();
    const subjectLower = subject.toLowerCase().trim();
    const classLower = className.toLowerCase().trim();
    const targetDate = date; // Already formatted as yyyy-MM-dd
    
    for (let i = 1; i < formData.length; i++) {
      const row = formData[i];
      if (!row[0]) continue;
      
      // Parse record date
      let recordDate = '';
      try {
        if (row[0] instanceof Date) {
          recordDate = Utilities.formatDate(row[0], 'Africa/Accra', 'yyyy-MM-dd');
        } else if (typeof row[0] === 'string') {
          const dateObj = new Date(row[0]);
          recordDate = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
        }
      } catch (e) {
        continue;
      }
      
      const recordTeacher = row[4] ? row[4].toString().toLowerCase().trim() : '';
      const recordSubject = row[5] ? row[5].toString().toLowerCase().trim() : '';
      const recordClasses = row[6] ? row[6].toString().toLowerCase().trim() : '';
      
      // Check for match
      if (recordDate === targetDate && 
          recordTeacher === teacherLower && 
          recordSubject === subjectLower &&
          recordClasses.includes(classLower)) {
        return true;
      }
    }
    
    return false;
  } catch (error) {
    Logger.log('Error checking attendance: ' + error);
    return false;
  }
}

/**
 * Clear false absences for a specific date range
 */
function clearFalseAbsencesByDateRange(startDate, endDate) {
  try {
    const start = new Date(startDate);
    const end = new Date(endDate);
    end.setDate(end.getDate() + 1); // Include end date
    
    Logger.log(`Clearing false absences from ${startDate} to ${endDate}`);
    
    const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
    const formDataSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    
    const absenceData = absenceSheet.getDataRange().getValues();
    const formData = formDataSheet.getDataRange().getValues();
    
    let deletedCount = 0;
    const deletions = [];
    
    // Delete from bottom to top
    for (let i = absenceData.length - 1; i >= 1; i--) {
      const row = absenceData[i];
      const absenceDate = row[0];
      const teacher = row[2];
      const subject = row[3];
      const className = row[4];
      const autoGenerated = row[7];
      
      if (autoGenerated === 'Yes') {
        let absenceDateObj;
        try {
          if (absenceDate instanceof Date) {
            absenceDateObj = absenceDate;
          } else {
            absenceDateObj = new Date(absenceDate);
          }
        } catch (e) {
          continue;
        }
        
        // Check if within date range
        if (absenceDateObj >= start && absenceDateObj < end) {
          const absenceDateFormatted = Utilities.formatDate(absenceDateObj, 'Africa/Accra', 'yyyy-MM-dd');
          const hasAttendance = checkAttendanceExists(formData, teacher, subject, className, absenceDateFormatted);
          
          if (hasAttendance) {
            absenceSheet.deleteRow(i + 1);
            deletedCount++;
            deletions.push(`${teacher} - ${subject} - ${className} (${absenceDateFormatted})`);
          }
        }
      }
    }
    
    Logger.log(`✅ Cleared ${deletedCount} false absences from ${startDate} to ${endDate}`);
    deletions.forEach(del => Logger.log(`   🗑️ ${del}`));
    
    return deletedCount;
    
  } catch (error) {
    Logger.log('Error clearing absences by date range: ' + error);
    return 0;
  }
}

    Logger.log('=== STARTING COMPREHENSIVE PAST ABSENCE SCAN ===');
    
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const timetableSheet = ss.getSheetByName('Timetable');
    const formDataSheet = ss.getSheetByName('FormData');
    const absenceSheet = ss.getSheetByName('Absences');
    
    if (!timetableSheet || !formDataSheet) {
      throw new Error('Required sheets (Timetable or FormData) not found');
    }
    
    // Create Absences sheet if it doesn't exist
    if (!absenceSheet) {
      const newAbsenceSheet = ss.insertSheet('Absences');
      newAbsenceSheet.getRange('A1:I1').setValues([[
        'Date', 'Time', 'Teacher', 'Subject', 'Class',
        'Scheduled Period', 'Status', 'Auto Generated', 'Reason'
      ]]);
      Logger.log('Created new Absences sheet');
    }
    
    // Get all timetable data
    const timetableData = timetableSheet.getDataRange().getValues();
    const timetableHeaders = timetableData[0];
    Logger.log(`Loaded ${timetableData.length - 1} timetable entries`);
    
    // Get all attendance data
    const attendanceData = formDataSheet.getDataRange().getValues();
    const attendanceHeaders = attendanceData[0];
    Logger.log(`Loaded ${attendanceData.length - 1} attendance records`);
    
    // Get date range from attendance data
    const dates = getUniqueDatesFromAttendance(attendanceData);
    Logger.log(`Scanning ${dates.length} dates: ${dates.join(', ')}`);
    
    let totalAbsencesRecorded = 0;
    let totalLessonsChecked = 0;
    
    // Process each date
    dates.forEach(date => {
      Logger.log(`\n📅 PROCESSING DATE: ${date}`);
      
      const dateAbsences = findAbsencesForDate(date, timetableData, attendanceData);
      const recordedCount = recordAbsencesForDate(date, dateAbsences);
      
      totalAbsencesRecorded += recordedCount;
      totalLessonsChecked += dateAbsences.length;
      
      Logger.log(`✅ Recorded ${recordedCount} absences for ${date}`);
    });
    
    // Generate summary report
    Logger.log('\n=== SCAN COMPLETED ===');
    Logger.log(`📊 SUMMARY:`);
    Logger.log(`   Dates processed: ${dates.length}`);
    Logger.log(`   Total lessons checked: ${totalLessonsChecked}`);
    Logger.log(`   Total absences recorded: ${totalAbsencesRecorded}`);
    
    // Send summary email
    sendAbsenceScanReport(dates.length, totalLessonsChecked, totalAbsencesRecorded);
    
    return {
      success: true,
      datesProcessed: dates.length,
      lessonsChecked: totalLessonsChecked,
      absencesRecorded: totalAbsencesRecorded,
      message: `Successfully scanned ${dates.length} dates and recorded ${totalAbsencesRecorded} absences`
    };
    
  } catch (error) {
    Logger.log('❌ ERROR in scanAndRecordPastAbsences: ' + error.toString());
    Logger.log('Stack: ' + error.stack);
    
    return {
      success: false,
      datesProcessed: 0,
      lessonsChecked: 0,
      absencesRecorded: 0,
      message: 'Failed to scan past absences: ' + error.message
    };
  }
}

// Get unique dates from attendance data
function getUniqueDatesFromAttendance(attendanceData) {
  const dates = new Set();
  
  for (let i = 1; i < attendanceData.length; i++) {
    const row = attendanceData[i];
    const dateValue = row[0]; // Date column
    
    if (dateValue) {
      try {
        let dateStr;
        if (dateValue instanceof Date) {
          dateStr = Utilities.formatDate(dateValue, 'Africa/Accra', 'yyyy-MM-dd');
        } else if (typeof dateValue === 'string') {
          const dateObj = new Date(dateValue);
          dateStr = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
        }
        
        if (dateStr && isValidDate(dateStr)) {
          dates.add(dateStr);
        }
      } catch (e) {
        // Skip invalid dates
        continue;
      }
    }
  }
  
  // Sort dates chronologically
  return Array.from(dates).sort();
}

// Check if date string is valid
function isValidDate(dateString) {
  const regex = /^\d{4}-\d{2}-\d{2}$/;
  if (!regex.test(dateString)) return false;
  
  const date = new Date(dateString);
  return date instanceof Date && !isNaN(date);
}

// Find absences for a specific date
function findAbsencesForDate(date, timetableData, attendanceData) {
  const absences = [];
  
  try {
    // Convert date to day of week (1=Monday, 7=Sunday)
    const dateObj = new Date(date);
    const dayOfWeek = getDayOfWeekFromDate(dateObj);
    
    Logger.log(`  Looking for timetable entries for ${getDayName(dayOfWeek)} (Day ${dayOfWeek})`);
    
    // Find all timetable entries for this day of week
    const dailyTimetable = [];
    for (let i = 1; i < timetableData.length; i++) {
      const row = timetableData[i];
      const timetableDay = parseInt(row[0], 10);
      
      if (timetableDay === dayOfWeek) {
        const teacher = row[3] ? row[3].toString().trim() : '';
        const subject = row[4] ? row[4].toString().trim() : '';
        const className = row[5] ? row[5].toString().trim() : '';
        const startTime = formatTimeForCalculation(row[1]);
        const endTime = formatTimeForCalculation(row[2]);
        
        if (teacher && subject && className) {
          dailyTimetable.push({
            teacher: teacher,
            subject: subject,
            className: className,
            startTime: startTime,
            endTime: endTime,
            period: `${startTime}-${endTime}`
          });
        }
      }
    }
    
    Logger.log(`  Found ${dailyTimetable.length} scheduled lessons for this day`);
    
    // Check each scheduled lesson for attendance
    dailyTimetable.forEach(lesson => {
      const hasAttendance = checkIfAttendanceExistsForLesson(attendanceData, date, lesson);
      
      if (!hasAttendance) {
        absences.push({
          date: date,
          teacher: lesson.teacher,
          subject: lesson.subject,
          className: lesson.className,
          period: lesson.period,
          startTime: lesson.startTime,
          endTime: lesson.endTime
        });
        
        Logger.log(`  ❌ ABSENCE: ${lesson.teacher} - ${lesson.subject} - ${lesson.className}`);
      } else {
        Logger.log(`  ✅ PRESENT: ${lesson.teacher} - ${lesson.subject} - ${lesson.className}`);
      }
    });
    
  } catch (error) {
    Logger.log(`  ERROR processing date ${date}: ` + error.toString());
  }
  
  return absences;
}

// Get day of week from date (1=Monday, 7=Sunday)
function getDayOfWeekFromDate(date) {
  const day = date.getDay(); // 0=Sunday, 1=Monday, ..., 6=Saturday
  return day === 0 ? 7 : day; // Convert to 1=Monday, 7=Sunday
}

// Check if attendance exists for a specific lesson
function checkIfAttendanceExistsForLesson(attendanceData, date, lesson) {
  for (let i = 1; i < attendanceData.length; i++) {
    const row = attendanceData[i];
    const recordDate = row[0];
    const teacher = row[4] ? row[4].toString().trim() : '';
    const subject = row[5] ? row[5].toString().trim() : '';
    const classes = row[6] ? row[6].toString().trim() : '';
    
    // Check if this record matches our lesson
    if (isSameDate(recordDate, date) && 
        teacher === lesson.teacher && 
        subject === lesson.subject &&
        classes.includes(lesson.className)) {
      return true;
    }
  }
  return false;
}

// Check if two dates are the same
function isSameDate(date1, dateString) {
  try {
    let date1Str;
    if (date1 instanceof Date) {
      date1Str = Utilities.formatDate(date1, 'Africa/Accra', 'yyyy-MM-dd');
    } else {
      const dateObj = new Date(date1);
      date1Str = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
    }
    return date1Str === dateString;
  } catch (e) {
    return false;
  }
}

// Record absences for a specific date
function recordAbsencesForDate(date, absences) {
  if (absences.length === 0) return 0;
  
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const absenceSheet = ss.getSheetByName('Absences');
  
  let recordedCount = 0;
  
  absences.forEach(absence => {
    // Check if absence is already recorded
    if (!isAbsenceAlreadyRecorded(absenceSheet, absence)) {
      // Record the absence
      const now = new Date();
      const recordTime = Utilities.formatDate(now, 'Africa/Accra', 'HH:mm');
      
      absenceSheet.appendRow([
        absence.date,
        recordTime,
        absence.teacher,
        absence.subject,
        absence.className,
        absence.period,
        'Absent',
        'Yes (Retroactive Scan)',
        'Automatically detected by retroactive scan'
      ]);
      
      recordedCount++;
      Logger.log(`    📝 Recorded absence: ${absence.teacher} - ${absence.subject}`);
    } else {
      Logger.log(`    ⏭️  Absence already recorded: ${absence.teacher} - ${absence.subject}`);
    }
  });
  
  return recordedCount;
}

// Check if absence is already recorded
function isAbsenceAlreadyRecorded(absenceSheet, absence) {
  const data = absenceSheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const recordDate = row[0];
    const teacher = row[2];
    const subject = row[3];
    const className = row[4];
    
    if (isSameDate(recordDate, absence.date) && 
        teacher === absence.teacher && 
        subject === absence.subject &&
        className === absence.className) {
      return true;
    }
  }
  return false;
}

// Function to scan a specific date range
function scanSpecificDateRange(startDate, endDate) {
  try {
    Logger.log(`=== SCANNING SPECIFIC DATE RANGE: ${startDate} to ${endDate} ===`);
    
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const timetableSheet = ss.getSheetByName('Timetable');
    const formDataSheet = ss.getSheetByName('FormData');
    const absenceSheet = ss.getSheetByName('Absences');
    
    if (!timetableSheet || !formDataSheet) {
      throw new Error('Required sheets not found');
    }
    
    const timetableData = timetableSheet.getDataRange().getValues();
    const attendanceData = formDataSheet.getDataRange().getValues();
    
    const start = new Date(startDate);
    const end = new Date(endDate);
    let currentDate = new Date(start);
    
    let totalAbsencesRecorded = 0;
    const processedDates = [];
    
    // Process each date in the range
    while (currentDate <= end) {
      const dateStr = Utilities.formatDate(currentDate, 'Africa/Accra', 'yyyy-MM-dd');
      
      // Skip weekends if desired (optional)
      const dayOfWeek = currentDate.getDay();
      if (dayOfWeek !== 0 && dayOfWeek !== 6) { // Skip Sunday (0) and Saturday (6)
        processedDates.push(dateStr);
        
        const absences = findAbsencesForDate(dateStr, timetableData, attendanceData);
        const recordedCount = recordAbsencesForDate(dateStr, absences);
        totalAbsencesRecorded += recordedCount;
        
        Logger.log(`Date: ${dateStr} - ${recordedCount} absences recorded`);
      }
      
      currentDate.setDate(currentDate.getDate() + 1);
    }
    
    Logger.log(`✅ Range scan completed: ${totalAbsencesRecorded} absences recorded for ${processedDates.length} dates`);
    
    return {
      success: true,
      datesProcessed: processedDates.length,
      absencesRecorded: totalAbsencesRecorded,
      message: `Scanned ${processedDates.length} dates and recorded ${totalAbsencesRecorded} absences`
    };
    
  } catch (error) {
    Logger.log('ERROR in scanSpecificDateRange: ' + error.toString());
    return {
      success: false,
      datesProcessed: 0,
      absencesRecorded: 0,
      message: 'Failed to scan date range: ' + error.message
    };
  }
}

function scanRecentDays(daysBack = 7) {
  const endDate = new Date();

/**
 * Scan for false absences within a date range
 */
function scanFalseAbsences(startDate, endDate, teacherFilter = '') {
  try {
    Logger.log(`=== SCANNING FALSE ABSENCES: ${startDate} to ${endDate} ===`);
    
    const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
    const formDataSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    
    if (!absenceSheet || !formDataSheet) {
      return {
        success: false,
        message: 'Required sheets not found',
        data: []
      };
    }
    
    const absenceData = absenceSheet.getDataRange().getValues();
    const formData = formDataSheet.getDataRange().getValues();
    
    const start = new Date(startDate);
    const end = new Date(endDate);
    end.setDate(end.getDate() + 1); // Include end date
    
    const falseAbsences = [];
    
    // Filter by teacher if specified
    const teacherFilterLower = teacherFilter.toLowerCase();
    
    for (let i = 1; i < absenceData.length; i++) {
      const row = absenceData[i];
      const absenceDate = row[0];
      const teacher = row[2] || '';
      const subject = row[3] || '';
      const className = row[4] || '';
      const autoGenerated = row[7] || '';
      const time = row[1] || '';
      
      // Skip if not auto-generated
      if (autoGenerated !== 'Yes') continue;
      
      // Apply teacher filter if specified
      if (teacherFilter && teacher.toLowerCase() !== teacherFilterLower) continue;
      
      // Check date
      let absenceDateObj;
      try {
        if (absenceDate instanceof Date) {
          absenceDateObj = absenceDate;
        } else {
          absenceDateObj = new Date(absenceDate);
        }
        
        if (absenceDateObj < start || absenceDateObj >= end) continue;
        
        const absenceDateFormatted = Utilities.formatDate(absenceDateObj, 'Africa/Accra', 'yyyy-MM-dd');
        
        // Check if attendance exists
        const hasAttendance = checkAttendanceExists(formData, teacher, subject, className, absenceDateFormatted);
        
        if (hasAttendance) {
          falseAbsences.push({
            date: absenceDateFormatted,
            teacher: teacher,
            subject: subject,
            className: className,
            time: time,
            rowIndex: i + 1
          });
        }
      } catch (e) {
        continue;
      }
    }
    
    Logger.log(`Found ${falseAbsences.length} potential false absences`);
    
    return {
      success: true,
      message: `Found ${falseAbsences.length} potential false absences`,
      data: falseAbsences
    };
    
  } catch (error) {
    Logger.log('ERROR in scanFalseAbsences: ' + error.toString());
    return {
      success: false,
      message: 'Error scanning false absences: ' + error.message,
      data: []
    };
  }
}

/**
 * Preview false absences for a date range (safe mode)
 */
function previewFalseAbsencesForRange(startDate, endDate, teacherFilter = '') {
  try {
    Logger.log(`=== PREVIEWING FALSE ABSENCES: ${startDate} to ${endDate} ===`);
    
    const result = scanFalseAbsences(startDate, endDate, teacherFilter);
    
    if (result.success && result.data.length > 0) {
      Logger.log(`Preview: ${result.data.length} false absences would be deleted`);
      
      // Group by teacher for detailed logging
      const byTeacher = {};
      result.data.forEach(absence => {
        if (!byTeacher[absence.teacher]) {
          byTeacher[absence.teacher] = [];
        }
        byTeacher[absence.teacher].push(absence);
      });
      
      Logger.log('Teachers affected: ' + Object.keys(byTeacher).join(', '));
    }
    
    return result;
    
  } catch (error) {
    Logger.log('ERROR in previewFalseAbsencesForRange: ' + error.toString());
    return {
      success: false,
      message: 'Error previewing false absences: ' + error.message,
      data: []
    };
  }
}

/**
 * Clear false absences for a specific date range with teacher filter
 */
function clearFalseAbsencesByDateRange(startDate, endDate, teacherFilter = '') {
  try {
    const start = new Date(startDate);
    const end = new Date(endDate);
    end.setDate(end.getDate() + 1); // Include end date
    
    Logger.log(`Clearing false absences from ${startDate} to ${endDate}, Teacher: ${teacherFilter || 'All'}`);
    
    const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
    const formDataSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    
    if (!absenceSheet || !formDataSheet) {
      return {
        success: false,
        deletedCount: 0,
        checkedCount: 0,
        message: 'Required sheets not found'
      };
    }
    
    const absenceData = absenceSheet.getDataRange().getValues();
    const formData = formDataSheet.getDataRange().getValues();
    
    let deletedCount = 0;
    let checkedCount = 0;
    const deletions = [];
    
    // Delete from bottom to top
    for (let i = absenceData.length - 1; i >= 1; i--) {
      const row = absenceData[i];
      const absenceDate = row[0];
      const teacher = row[2] || '';
      const subject = row[3] || '';
      const className = row[4] || '';
      const autoGenerated = row[7] || '';
      
      // Only check auto-generated records
      if (autoGenerated === 'Yes') {
        // Apply teacher filter if specified
        if (teacherFilter && teacher !== teacherFilter) continue;
        
        checkedCount++;
        
        let absenceDateObj;
        try {
          if (absenceDate instanceof Date) {
            absenceDateObj = absenceDate;
          } else {
            absenceDateObj = new Date(absenceDate);
          }
          
          // Check if within date range
          if (absenceDateObj >= start && absenceDateObj < end) {
            const absenceDateFormatted = Utilities.formatDate(absenceDateObj, 'Africa/Accra', 'yyyy-MM-dd');
            const hasAttendance = checkAttendanceExists(formData, teacher, subject, className, absenceDateFormatted);
            
            if (hasAttendance) {
              absenceSheet.deleteRow(i + 1);
              deletedCount++;
              deletions.push({
                teacher: teacher,
                subject: subject,
                className: className,
                date: absenceDateFormatted
              });
            }
          }
        } catch (e) {
          continue;
        }
      }
    }
    
    // Generate detailed report
    if (deletedCount > 0) {
      Logger.log(`✅ Cleared ${deletedCount} false absences`);
      
      // Send email notification if many records were deleted
      if (deletedCount > 10) {
        sendDeletionReport(deletions, startDate, endDate, teacherFilter);
      }
    }
    
    return {
      success: true,
      deletedCount: deletedCount,
      checkedCount: checkedCount,
      message: `Cleared ${deletedCount} false absence records from ${startDate} to ${endDate}`
    };
    
  } catch (error) {
    Logger.log('ERROR in clearFalseAbsencesByDateRange: ' + error.toString());
    return {
      success: false,
      deletedCount: 0,
      checkedCount: 0,
      message: 'Failed to clear false absences: ' + error.message
    };
  }
}

/**
 * Export false absence report
 */
function exportFalseAbsenceReport(startDate, endDate, teacherFilter = '') {
  try {
    const result = scanFalseAbsences(startDate, endDate, teacherFilter);
    
    if (!result.success || result.data.length === 0) {
      return null;
    }
    
    // Create a temporary spreadsheet for export
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const exportSheetName = `False Absence Report ${startDate} to ${endDate}`;
    
    // Check if export sheet already exists
    let exportSheet = ss.getSheetByName(exportSheetName);
    if (exportSheet) {
      ss.deleteSheet(exportSheet);
    }
    
    // Create new sheet
    exportSheet = ss.insertSheet(exportSheetName);
    
    // Set headers
    exportSheet.getRange('A1:H1').setValues([[
      'Date', 'Teacher', 'Subject', 'Class', 'Time',
      'Absence Type', 'Status', 'Attendance Verified'
    ]]);
    
    // Format headers
    exportSheet.getRange('A1:H1').setFontWeight('bold').setBackground('#4a86e8').setFontColor('white');
    
    // Add data
    const exportData = result.data.map(absence => [
      absence.date,
      absence.teacher,
      absence.subject,
      absence.className,
      absence.time,
      'Auto Generated',
      'False Absence',
      'Present'
    ]);
    
    if (exportData.length > 0) {
      exportSheet.getRange(2, 1, exportData.length, 8).setValues(exportData);
      
      // Auto-resize columns
      for (let i = 1; i <= 8; i++) {
        exportSheet.autoResizeColumn(i);
      }
      
      // Add summary
      const summaryRow = exportData.length + 3;
      exportSheet.getRange(`A${summaryRow}:H${summaryRow}`).setValues([[
        'SUMMARY', '', '', '', '', '', '', ''
      ]]);
      exportSheet.getRange(`A${summaryRow}:H${summaryRow}`).setFontWeight('bold').setBackground('#f6b26b');
      
      exportSheet.getRange(`A${summaryRow + 1}`).setValue('Total False Absences:');
      exportSheet.getRange(`B${summaryRow + 1}`).setValue(exportData.length);
      
      exportSheet.getRange(`A${summaryRow + 2}`).setValue('Date Range:');
      exportSheet.getRange(`B${summaryRow + 2}`).setValue(`${startDate} to ${endDate}`);
      
      exportSheet.getRange(`A${summaryRow + 3}`).setValue('Teacher Filter:');
      exportSheet.getRange(`B${summaryRow + 3}`).setValue(teacherFilter || 'All Teachers');
      
      exportSheet.getRange(`A${summaryRow + 4}`).setValue('Generated On:');
      exportSheet.getRange(`B${summaryRow + 4}`).setValue(new Date().toLocaleString());
    }
    
    // Create PDF
    const url = exportAsPDF(exportSheet);
    
    // Delete the temporary sheet after a delay
    Utilities.sleep(5000); // Wait 5 seconds
    ss.deleteSheet(exportSheet);
    
    return url;
    
  } catch (error) {
    Logger.log('ERROR exporting false absence report: ' + error.toString());
    return null;
  }
}

// Fixed function to identify duplicate attendance records with better error handling
function getDuplicateAttendanceRecords() {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    const lastRow = sheet.getLastRow();
    
    if (lastRow < 2) {
      return { success: true, message: "No records found to check", duplicates: [] };
    }
    
    // Get all data with row numbers
    const data = sheet.getRange(1, 1, lastRow, sheet.getLastColumn()).getValues();
    const headers = data[0];
    
    const recordMap = new Map();
    const duplicates = [];
    
    // Identify duplicates (same teacher, subject, class, and date)
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const rowNumber = i + 1; // Actual row number in sheet
      
      // Safe date parsing with error handling
      let recordDate = 'No Date';
      try {
        if (row[0] && row[0] instanceof Date) {
          recordDate = Utilities.formatDate(row[0], Session.getScriptTimeZone(), 'yyyy-MM-dd');
        } else if (row[0] && typeof row[0] === 'string' && row[0].trim() !== '') {
          const dateObj = new Date(row[0]);
          if (!isNaN(dateObj.getTime())) {
            recordDate = Utilities.formatDate(dateObj, Session.getScriptTimeZone(), 'yyyy-MM-dd');
          }
        }
      } catch (e) {
        recordDate = 'Invalid Date';
      }
      
      const teacher = row[4] ? row[4].toString().trim() : 'No Teacher';
      const subject = row[5] ? row[5].toString().trim() : 'No Subject';
      const classNames = row[6] ? row[6].toString().trim() : 'No Class';
      const submissionTime = row[1] ? row[1].toString() : 'No Time';
      const topic = row[8] ? row[8].toString().trim() : 'No Topic';
      
      // Safe timestamp creation
      let timestamp;
      try {
        if (row[0] && row[1]) {
          const dateTimeStr = recordDate + ' ' + submissionTime;
          timestamp = new Date(dateTimeStr);
          if (isNaN(timestamp.getTime())) {
            timestamp = new Date(); // Fallback to current time
          }
        } else {
          timestamp = new Date(); // Fallback to current time
        }
      } catch (e) {
        timestamp = new Date(); // Fallback to current time
      }
      
      const key = `${recordDate}_${teacher}_${subject}_${classNames}`;
      
      if (recordMap.has(key)) {
        // This is a duplicate
        const existingRecord = recordMap.get(key);
        duplicates.push({
          key: key,
          records: [
            {
              rowNumber: existingRecord.rowNumber,
              date: existingRecord.date,
              time: existingRecord.time,
              teacher: existingRecord.teacher,
              subject: existingRecord.subject,
              classNames: existingRecord.classNames,
              topic: existingRecord.topic,
              timestamp: existingRecord.timestamp
            },
            {
              rowNumber: rowNumber,
              date: recordDate,
              time: submissionTime,
              teacher: teacher,
              subject: subject,
              classNames: classNames,
              topic: topic,
              timestamp: timestamp
            }
          ]
        });
        
        // Update the map to track all occurrences for this key
        recordMap.get(key).occurrences.push({
          rowNumber: rowNumber,
          date: recordDate,
          time: submissionTime,
          teacher: teacher,
          subject: subject,
          classNames: classNames,
          topic: topic,
          timestamp: timestamp
        });
      } else {
        // First occurrence of this record
        recordMap.set(key, {
          rowNumber: rowNumber,
          date: recordDate,
          time: submissionTime,
          teacher: teacher,
          subject: subject,
          classNames: classNames,
          topic: topic,
          timestamp: timestamp,
          occurrences: [{
            rowNumber: rowNumber,
            date: recordDate,
            time: submissionTime,
            teacher: teacher,
            subject: subject,
            classNames: classNames,
            topic: topic,
            timestamp: timestamp
          }]
        });
      }
    }
    
    // For records with more than 2 occurrences, group them properly
    const allDuplicates = [];
    recordMap.forEach((record, key) => {
      if (record.occurrences.length > 1) {
        allDuplicates.push({
          key: key,
          records: record.occurrences
        });
      }
    });
    
    // Sort duplicates by date (newest first) for easier review
    allDuplicates.sort((a, b) => {
      try {
        const dateA = new Date(a.records[0].date);
        const dateB = new Date(b.records[0].date);
        return dateB - dateA;
      } catch (e) {
        return 0;
      }
    });
    
    Logger.log(`Found ${allDuplicates.length} sets of duplicate records`);
    
    return {
      success: true,
      message: `Found ${allDuplicates.length} sets of duplicate records`,
      duplicates: allDuplicates,
      totalRecords: lastRow - 1
    };
    
  } catch (error) {
    Logger.log('Error finding duplicates: ' + error.message);
    return {
      success: false,
      message: 'Error: ' + error.message,
      duplicates: [],
      totalRecords: 0
    };
  }
}

// Function to keep only the latest record from duplicates
function keepLatestRecordsOnly() {
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('FormData');
    const lastRow = sheet.getLastRow();
    
    if (lastRow < 2) {
      return { success: true, message: "No records found to process" };
    }
    
    const data = sheet.getRange(1, 1, lastRow, sheet.getLastColumn()).getValues();
    const recordMap = new Map();
    const rowsToDelete = [];
    
    // Identify which records to keep (latest ones)
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const rowNumber = i + 1;
      
      const recordDate = row[0] ? Utilities.formatDate(new Date(row[0]), Session.getScriptTimeZone(), 'yyyy-MM-dd') : '';
      const teacher = row[4] ? row[4].toString().trim() : '';
      const subject = row[5] ? row[5].toString().trim() : '';
      const classNames = row[6] ? row[6].toString().trim() : '';
      const submissionTime = row[1] ? new Date(row[1]) : new Date();
      
      const key = `${recordDate}_${teacher}_${subject}_${classNames}`;
      
      if (recordMap.has(key)) {
        // Compare timestamps and keep the latest
        const existing = recordMap.get(key);
        if (submissionTime > existing.timestamp) {
          // Current record is newer, delete the old one
          rowsToDelete.push(existing.rowNumber);
          recordMap.set(key, { rowNumber: rowNumber, timestamp: submissionTime });
        } else {
          // Existing record is newer, delete current one
          rowsToDelete.push(rowNumber);
        }
      } else {
        // First occurrence
        recordMap.set(key, { rowNumber: rowNumber, timestamp: submissionTime });
      }
    }
    
    // Delete the older records
    let deletedCount = 0;
    rowsToDelete.sort((a, b) => b - a).forEach(rowNumber => {
      sheet.deleteRow(rowNumber);
      deletedCount++;
    });
    
    return {
      success: true,
      message: `Automatically kept latest records. Deleted ${deletedCount} older duplicate records.`,
      deletedCount: deletedCount
    };
    
  } catch (error) {
    Logger.log('Error in keepLatestRecordsOnly: ' + error.message);
    return {
      success: false,
      message: 'Error: ' + error.message,
      deletedCount: 0
    };
  }
}

/**
 * Scan and record absences for a specific date range
 */
function scanAndRecordAbsencesForDateRange(startDate, endDate) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const timetableSheet = ss.getSheetByName('Timetable');
    const formDataSheet = ss.getSheetByName('FormData');
    const absenceSheet = ss.getSheetByName('Absences');
    
    if (!timetableSheet || !formDataSheet) {
      throw new Error('Required sheets not found');
    }
    
    // Ensure Absences sheet exists
    if (!absenceSheet) {
      const newAbsenceSheet = ss.insertSheet('Absences');
      newAbsenceSheet.getRange('A1:I1').setValues([[
        'Date', 'Time', 'Teacher', 'Subject', 'Class',
        'Scheduled Period', 'Status', 'Auto Generated', 'Reason'
      ]]);
      Logger.log('Created new Absences sheet');
    }
    
    const timetableData = timetableSheet.getDataRange().getValues();
    const attendanceData = formDataSheet.getDataRange().getValues();
    
    const start = new Date(startDate);
    const end = new Date(endDate);
    let currentDate = new Date(start);
    
    let totalAbsencesRecorded = 0;
    let totalLessonsChecked = 0;
    const processedDates = [];
    
    Logger.log(`Scanning date range: ${startDate} to ${endDate}`);
    
    // Process each date in the range
    while (currentDate <= end) {
      const dateStr = Utilities.formatDate(currentDate, 'Africa/Accra', 'yyyy-MM-dd');
      const dayOfWeek = currentDate.getDay();
      
      // Only process weekdays (Monday to Friday)
      if (dayOfWeek >= 1 && dayOfWeek <= 5) {
        processedDates.push(dateStr);
        
        Logger.log(`\n📅 Processing ${dateStr} (${getDayName(dayOfWeek)})`);
        
        const absences = findAbsencesForSpecificDate(dateStr, timetableData, attendanceData);
        const recordedCount = recordAbsencesForSpecificDate(dateStr, absences);
        
        totalAbsencesRecorded += recordedCount;
        totalLessonsChecked += absences.length;
        
        Logger.log(`   Checked ${absences.length} lessons, recorded ${recordedCount} absences`);
      } else {
        Logger.log(`⏭️  Skipping ${dateStr} (weekend)`);
      }
      
      currentDate.setDate(currentDate.getDate() + 1);
    }
    
    const result = {
      success: true,
      dateRange: `${startDate} to ${endDate}`,
      datesProcessed: processedDates.length,
      lessonsChecked: totalLessonsChecked,
      absencesRecorded: totalAbsencesRecorded,
      message: `Successfully processed ${processedDates.length} dates and recorded ${totalAbsencesRecorded} absences for the gap period`
    };
    
    Logger.log(`\n📊 FINAL RESULT: ${JSON.stringify(result)}`);
    return result;
    
  } catch (error) {
    Logger.log('ERROR in scanAndRecordAbsencesForDateRange: ' + error.toString());
    throw error;
  }
}

/**
 * Find absences for a specific date by comparing timetable with attendance
 */
function findAbsencesForSpecificDate(dateStr, timetableData, attendanceData) {
  const absences = [];
  
  try {
    const dateObj = new Date(dateStr);
    const dayOfWeek = getDayOfWeekFromDate(dateObj);
    
    Logger.log(`   Looking for timetable entries for ${getDayName(dayOfWeek)}`);
    
    // Find all timetable entries for this day of week
    const dailyTimetable = [];
    for (let i = 1; i < timetableData.length; i++) {
      const row = timetableData[i];
      const timetableDay = parseInt(row[0], 10);
      
      if (timetableDay === dayOfWeek) {
        const teacher = row[3] ? row[3].toString().trim() : '';
        const subject = row[4] ? row[4].toString().trim() : '';
        const className = row[5] ? row[5].toString().trim() : '';
        const startTime = formatTimeForCalculation(row[1]);
        const endTime = formatTimeForCalculation(row[2]);
        
        if (teacher && subject && className) {
          dailyTimetable.push({
            teacher: teacher,
            subject: subject,
            className: className,
            startTime: startTime,
            endTime: endTime,
            period: `${startTime}-${endTime}`
          });
        }
      }
    }
    
    Logger.log(`   Found ${dailyTimetable.length} scheduled lessons`);
    
    // Check each scheduled lesson for attendance
    dailyTimetable.forEach(lesson => {
      const hasAttendance = checkIfAttendanceExists(attendanceData, dateStr, lesson);
      
      if (!hasAttendance) {
        absences.push({
          date: dateStr,
          teacher: lesson.teacher,
          subject: lesson.subject,
          className: lesson.className,
          period: lesson.period,
          startTime: lesson.startTime,
          endTime: lesson.endTime
        });
        
        Logger.log(`   ❌ ABSENCE: ${lesson.teacher} - ${lesson.subject} - ${lesson.className}`);
      } else {
        Logger.log(`   ✅ PRESENT: ${lesson.teacher} - ${lesson.subject} - ${lesson.className}`);
      }
    });
    
  } catch (error) {
    Logger.log(`   ERROR processing date ${dateStr}: ` + error.toString());
  }
  
  return absences;
}

/**
 * Check if attendance exists for a specific lesson
 */
function checkIfAttendanceExists(attendanceData, dateStr, lesson) {
  for (let i = 1; i < attendanceData.length; i++) {
    const row = attendanceData[i];
    if (!row[0]) continue;
    
    // Parse record date
    let recordDate = '';
    try {
      if (row[0] instanceof Date) {
        recordDate = Utilities.formatDate(row[0], 'Africa/Accra', 'yyyy-MM-dd');
      } else if (typeof row[0] === 'string') {
        const dateObj = new Date(row[0]);
        recordDate = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
      }
    } catch (e) {
      continue;
    }
    
    const teacher = row[4] ? row[4].toString().trim() : '';
    const subject = row[5] ? row[5].toString().trim() : '';
    const classes = row[6] ? row[6].toString().trim() : '';
    
    // Check for match
    if (recordDate === dateStr && 
        teacher === lesson.teacher && 
        subject === lesson.subject &&
        classes.includes(lesson.className)) {
      return true;
    }
  }
  return false;
}

/**
 * Record absences for a specific date
 */
function recordAbsencesForSpecificDate(dateStr, absences) {
  if (absences.length === 0) return 0;
  
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const absenceSheet = ss.getSheetByName('Absences');
  
  let recordedCount = 0;
  
  absences.forEach(absence => {
    // Check if absence is already recorded
    if (!isAbsenceAlreadyRecorded(absenceSheet, absence)) {
      // Record the absence
      const now = new Date();
      const recordTime = Utilities.formatDate(now, 'Africa/Accra', 'HH:mm');
      
      absenceSheet.appendRow([
        absence.date,
        recordTime,
        absence.teacher,
        absence.subject,
        absence.className,
        absence.period,
        'Absent',
        'Yes (Gap Fill)',
        'Automatically recorded during gap filling process'
      ]);
      
      recordedCount++;
      Logger.log(`      📝 Recorded: ${absence.teacher} - ${absence.subject}`);
    } else {
      Logger.log(`      ⏭️  Already recorded: ${absence.teacher} - ${absence.subject}`);
    }
  });
  
  return recordedCount;
}

/**
 * Check if absence is already recorded
 */
function isAbsenceAlreadyRecorded(absenceSheet, absence) {
  const data = absenceSheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const recordDate = row[0];
    const teacher = row[2];
    const subject = row[3];
    const className = row[4];
    
    let recordDateStr = '';
    try {
      if (recordDate instanceof Date) {
        recordDateStr = Utilities.formatDate(recordDate, 'Africa/Accra', 'yyyy-MM-dd');
      } else if (typeof recordDate === 'string') {
        const dateObj = new Date(recordDate);
        recordDateStr = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
      }
    } catch (e) {
      continue;
    }
    
    if (recordDateStr === absence.date && 
        teacher === absence.teacher && 
        subject === absence.subject &&
        className === absence.className) {
      return true;
    }
  }
  return false;
}

/**
 * Get day name from day number
 */
function getDayName(dayOfWeek) {
  const dayNames = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'];
  return dayNames[dayOfWeek] || `Day ${dayOfWeek}`;
}

/**
 * Get day of week from date (1=Monday, 7=Sunday)
 */
function getDayOfWeekFromDate(date) {
  const day = date.getDay(); // 0=Sunday, 1=Monday, ..., 6=Saturday
  return day === 0 ? 7 : day; // Convert to 1=Monday, 7=Sunday
}

function getOutstandingAbsences(teacherName) {
  try {
    Logger.log(`=== Checking outstanding absences for: ${teacherName} ===`);
    
    const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
    const remedialSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('RemedialLessons');
    
    if (!absenceSheet) {
      Logger.log('ERROR: No Absences sheet found');
      return [];
    }
    
    const absenceData = absenceSheet.getDataRange().getValues();
    const remedialData = remedialSheet ? remedialSheet.getDataRange().getValues() : [[]];
    
    Logger.log('Total absence records: ' + (absenceData.length - 1));
    
    const outstandingAbsences = [];
    
    // Get current date - DON'T FILTER BY 30 DAYS for now
    const today = new Date();
    
    Logger.log('Today: ' + Utilities.formatDate(today, 'Africa/Accra', 'yyyy-MM-dd'));
    Logger.log('Checking ALL absences (no date filter)');
    
    for (let i = 1; i < absenceData.length; i++) {
      const row = absenceData[i];
      
      // Extract data
      const rawDate = row[0];
      const teacher = row[2] ? row[2].toString().trim() : '';
      const subject = row[3] ? row[3].toString().trim() : '';
      const className = row[4] ? row[4].toString().trim() : '';
      const autoGenerated = row[7] ? row[7].toString().trim() : '';
      const status = row[6] ? row[6].toString().trim() : '';
      
      // Skip if not this teacher
      if (teacher !== teacherName) continue;
      
      // Check if this is auto-generated - be flexible with the value
      // Accept: "Yes", "Yes (Manual Recording)", "Yes (Gap Fill)"
      const isAutoGenerated = autoGenerated.includes('Yes') || autoGenerated.includes('YES');
      
      if (!isAutoGenerated) {
        Logger.log(`Skipping - not auto-generated: "${autoGenerated}"`);
        continue;
      }
      
      // Skip if already resolved (contains remedial/made up/verified)
      const statusUpper = status.toUpperCase();
      if (statusUpper.includes('REMEDIAL') || 
          statusUpper.includes('MADE UP') || 
          statusUpper.includes('CLEARED') || 
          statusUpper.includes('VERIFIED') ||
          statusUpper.includes('COMPLETED')) {
        Logger.log(`Skipping - already resolved: "${status}"`);
        continue;
      }
      
      // Parse date
      let absenceDateStr = '';
      try {
        if (rawDate instanceof Date) {
          absenceDateStr = Utilities.formatDate(rawDate, 'Africa/Accra', 'yyyy-MM-dd');
        } else if (typeof rawDate === 'string') {
          const dateObj = new Date(rawDate);
          absenceDateStr = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
        }
      } catch (e) {
        Logger.log(`Date parsing error: ${e.message} for value: ${rawDate}`);
        continue;
      }
      
      if (!absenceDateStr) {
        Logger.log(`Invalid date: ${rawDate}`);
        continue;
      }
      
      Logger.log(`Processing absence: ${absenceDateStr} - ${subject} - ${className}`);
      
      // Check if remedial already exists
      let hasRemedial = false;
      if (remedialSheet && remedialData.length > 1) {
        for (let j = 1; j < remedialData.length; j++) {
          const remedialRow = remedialData[j];
          const remedialTeacher = remedialRow[1] ? remedialRow[1].toString().trim() : '';
          const remedialStatus = remedialRow[11] ? remedialRow[11].toString().trim().toUpperCase() : '';
          const remedialSubject = remedialRow[2] ? remedialRow[2].toString().trim() : '';
          const remedialClass = remedialRow[3] ? remedialRow[3].toString().trim() : '';
          
          if (remedialTeacher === teacherName && 
              remedialSubject === subject && 
              remedialClass === className &&
              ['SCHEDULED', 'COMPLETED', 'VERIFIED'].includes(remedialStatus)) {
            
            // Check dates match
            const remedialOriginalDate = remedialRow[0];
            let remedialDateStr = '';
            try {
              if (remedialOriginalDate instanceof Date) {
                remedialDateStr = Utilities.formatDate(remedialOriginalDate, 'Africa/Accra', 'yyyy-MM-dd');
              } else if (typeof remedialOriginalDate === 'string') {
                remedialDateStr = remedialOriginalDate.toString().trim();
              }
            } catch (e) {
              continue;
            }
            
            if (remedialDateStr === absenceDateStr) {
              hasRemedial = true;
              Logger.log(`  Found existing remedial with status: ${remedialStatus}`);
              break;
            }
          }
        }
      }
      
      if (!hasRemedial) {
        outstandingAbsences.push({
          date: absenceDateStr,
          subject: subject,
          className: className,
          time: row[1] ? row[1].toString().trim() : '',
          reason: row[8] ? row[8].toString().trim() : '',
          scheduledPeriod: row[5] ? row[5].toString().trim() : '',
          status: status,
          autoGenerated: autoGenerated,
          rowIndex: i + 1
        });
        
        Logger.log(`✓ ADDED outstanding absence: ${absenceDateStr} - ${subject} - ${className}`);
      }
    }
    
    // Sort by date (most recent first)
    outstandingAbsences.sort((a, b) => new Date(b.date) - new Date(a.date));
    
    Logger.log(`Total outstanding absences found for ${teacherName}: ${outstandingAbsences.length}`);
    
    // Log all found absences for debugging
    if (outstandingAbsences.length > 0) {
      Logger.log('Found absences:');
      outstandingAbsences.forEach((abs, idx) => {
        Logger.log(`${idx + 1}. ${abs.date} - ${abs.subject} (${abs.className}) - Status: "${abs.status}"`);
      });
    }
    
    return outstandingAbsences;
    
  } catch (error) {
    Logger.log('ERROR in getOutstandingAbsences: ' + error.message);
    Logger.log('Stack trace: ' + error.stack);
    return [];
  }
}

// Schedule a remedial lesson
function scheduleRemedialLesson(data) {
  try {
    Logger.log('=== SCHEDULE REMEDIAL LESSON - START ===');
    Logger.log('Input data received:', JSON.stringify(data));
    
    // 1. Initialize sheet with debugging
    Logger.log('1. Initializing remedial sheet...');
    initializeRemedialSheet();
    Logger.log('✓ Remedial sheet initialized');
    
    // 2. Get remedial sheet
    const remedialSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('RemedialLessons');
    if (!remedialSheet) {
      Logger.log('✗ ERROR: RemedialLessons sheet not found after initialization');
      throw new Error('RemedialLessons sheet not found');
    }
    Logger.log('✓ Remedial sheet found');
    
    // 3. Validate required fields
    Logger.log('2. Validating input data...');
    if (!data.teacher) {
      throw new Error('Teacher name is required');
    }
    if (!data.originalDate) {
      throw new Error('Original absence date is required');
    }
    if (!data.subject) {
      throw new Error('Subject is required');
    }
    if (!data.className) {
      throw new Error('Class name is required');
    }
    if (!data.remedialDate) {
      throw new Error('Remedial date is required');
    }
    if (!data.remedialTime) {
      throw new Error('Remedial time is required');
    }
    Logger.log('✓ Input data validated');
    
    Logger.log(`Teacher: ${data.teacher}`);
    Logger.log(`Original Date: ${data.originalDate}`);
    Logger.log(`Subject: ${data.subject}`);
    Logger.log(`Class: ${data.className}`);
    Logger.log(`Remedial Date: ${data.remedialDate}`);
    Logger.log(`Remedial Time: ${data.remedialTime}`);
    Logger.log(`Topic: ${data.topic || '(not specified)'}`);
    Logger.log(`Location: ${data.location || 'Classroom'}`);
    
    // 4. Validate remedial date is not in the past (allow same day)
    Logger.log('3. Validating remedial date...');
    const today = new Date();
    const remedialDateObj = new Date(data.remedialDate);
    
    // Set both dates to midnight for fair comparison
    const todayMidnight = new Date(today.getFullYear(), today.getMonth(), today.getDate());
    const remedialMidnight = new Date(remedialDateObj.getFullYear(), remedialDateObj.getMonth(), remedialDateObj.getDate());
    
    Logger.log(`Today (midnight): ${todayMidnight}`);
    Logger.log(`Remedial date (midnight): ${remedialMidnight}`);
    
    // Allow remedial lessons for today and future dates
    if (remedialMidnight < todayMidnight) {
      throw new Error('Remedial date cannot be in the past. Please select today or a future date.');
    }
    Logger.log('✓ Remedial date validated');
    
    // 5. Check for existing remedial lessons
    Logger.log('4. Checking for existing remedial lessons...');
    const existingData = remedialSheet.getDataRange().getValues();
    Logger.log(`Total rows in remedial sheet: ${existingData.length}`);
    
    let duplicateFound = false;
    for (let i = 1; i < existingData.length; i++) {
      const row = existingData[i];
      const rowTeacher = row[1];
      const rowOriginalDate = row[0];
      const rowSubject = row[2];
      const rowClassName = row[3];
      const rowStatus = row[11];
      
      // Debug each row if it matches teacher
      if (rowTeacher === data.teacher) {
        Logger.log(`  Row ${i}: ${rowTeacher} - ${rowOriginalDate} - ${rowSubject} - ${rowClassName} - ${rowStatus}`);
      }
      
      if (rowTeacher === data.teacher && 
          rowOriginalDate === data.originalDate && 
          rowSubject === data.subject && 
          rowClassName === data.className &&
          ['Scheduled', 'Completed'].includes(rowStatus)) {
        duplicateFound = true;
        Logger.log(`✗ DUPLICATE FOUND at row ${i}: Status = ${rowStatus}`);
        throw new Error('A remedial lesson is already scheduled or in progress for this absence');
      }
    }
    Logger.log('✓ No duplicate lessons found');
    
    // 6. Prepare data for insertion
    Logger.log('5. Preparing data for insertion...');
    const now = new Date();
    const nowStr = Utilities.formatDate(now, 'Africa/Accra', 'yyyy-MM-dd HH:mm:ss');
    
    const newRowData = [
      data.originalDate,           // A: Original Absence Date (should already be yyyy-MM-dd)
      data.teacher,                // B: Teacher
      data.subject,                // C: Subject
      data.className,              // D: Class
      data.remedialDate,           // E: Remedial Date (ensure it's yyyy-MM-dd)
      data.remedialTime,           // F: Remedial Time
      data.duration || 2,          // G: Duration (periods)
      data.topic || '',            // H: Topic to cover
      data.location || 'Classroom', // I: Location
      '',                          // J: Photo URL (empty initially)
      '',                          // K: GPS (empty initially)
      'Scheduled',                 // L: Status
      '',                          // M: Verified By
      '',                          // N: Verified Date
      `Scheduled on ${nowStr}`,    // O: Notes
      '',                          // P: Row Index (from Absences sheet)
      nowStr,                      // Q: Date Created
      nowStr                       // R: Last Updated
    ];
    
    Logger.log('New row data prepared:', newRowData);
    
    // 7. Append to sheet
    Logger.log('6. Appending to sheet...');
    const lastRowBefore = remedialSheet.getLastRow();
    remedialSheet.appendRow(newRowData);
    const lastRowAfter = remedialSheet.getLastRow();
    
    Logger.log(`✓ Row appended. Last row before: ${lastRowBefore}, after: ${lastRowAfter}`);
    
    // 8. Also update the Absences sheet status
    Logger.log('7. Updating Absences sheet...');
    try {
      const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
      if (absenceSheet) {
        const absenceData = absenceSheet.getDataRange().getValues();
        let absenceFound = false;
        
        Logger.log(`Searching Absences sheet for: ${data.teacher}, ${data.originalDate}, ${data.subject}, ${data.className}`);
        
        for (let i = 1; i < absenceData.length; i++) {
          const row = absenceData[i];
          const absenceDate = row[0];
          const teacher = row[2];
          const subject = row[3];
          const className = row[4];
          
          let absenceDateStr;
          try {
            if (absenceDate instanceof Date) {
              absenceDateStr = Utilities.formatDate(absenceDate, 'Africa/Accra', 'yyyy-MM-dd');
            } else {
              const dateObj = new Date(absenceDate);
              absenceDateStr = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
            }
          } catch (e) {
            continue;
          }
          
          Logger.log(`  Row ${i}: ${teacher} - ${absenceDateStr} - ${subject} - ${className}`);
          
          if (teacher === data.teacher && 
              absenceDateStr === data.originalDate && 
              subject === data.subject && 
              className === data.className) {
            // Update status to show remedial is scheduled
            absenceSheet.getRange(i + 1, 7).setValue('Remedial Scheduled'); // Column G: Status
            absenceSheet.getRange(i + 1, 8).setNote(`Remedial scheduled for ${data.remedialDate} ${data.remedialTime}`);
            absenceFound = true;
            Logger.log(`✓ Updated Absences sheet at row ${i + 1}`);
            break;
          }
        }
        
        if (!absenceFound) {
          Logger.log('⚠️ No matching absence record found in Absences sheet');
        }
      } else {
        Logger.log('⚠️ Absences sheet not found - skipping update');
      }
    } catch (updateError) {
      Logger.log('⚠️ Could not update Absences sheet: ' + updateError.message);
      // Continue even if this fails
    }
    
    // 9. Send notification to teacher (optional)
    Logger.log('8. Sending notification...');
    try {
      sendRemedialScheduledNotification(data.teacher, data.remedialDate, data.remedialTime);
      Logger.log('✓ Notification sent');
    } catch (notifyError) {
      Logger.log('⚠️ Could not send notification: ' + notifyError.message);
    }
    
    // 10. Final success
    Logger.log('=== SCHEDULE REMEDIAL LESSON - COMPLETE ===');
    
    return {
      success: true,
      message: '✅ Remedial lesson scheduled successfully!',
      nextStep: 'Please conduct the remedial lesson at the scheduled time and submit attendance through the "Submit Remedial Attendance" section',
      debug: {
        teacher: data.teacher,
        originalDate: data.originalDate,
        remedialDate: data.remedialDate,
        remedialTime: data.remedialTime,
        subject: data.subject,
        className: data.className,
        rowAdded: lastRowAfter,
        timestamp: nowStr
      }
    };
    
  } catch (error) {
    Logger.log('=== SCHEDULE REMEDIAL LESSON - ERROR ===');
    Logger.log('Error type: ' + typeof error);
    Logger.log('Error message: ' + error.message);
    Logger.log('Error stack: ' + error.stack);
    Logger.log('Data received: ' + JSON.stringify(data));
    
    return {
      success: false,
      message: '❌ Failed to schedule remedial lesson: ' + error.message,
      debug: {
        error: error.message,
        stack: error.stack,
        dataReceived: data
      }
    };
  }
}

          if (remedialDateCell instanceof Date) {
            remedialDateStr = Utilities.formatDate(remedialDateCell, 'Africa/Accra', 'yyyy-MM-dd');
          } else if (typeof remedialDateCell === 'string') {
            // Clean the string
            const cleanDateStr = remedialDateCell.toString().trim();
            if (cleanDateStr.match(/^\d{4}-\d{2}-\d{2}$/)) {
              remedialDateStr = cleanDateStr;
            } else {
              const dateObj = new Date(cleanDateStr);
              if (!isNaN(dateObj.getTime())) {
                remedialDateStr = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
              }
            }
          } else if (remedialDateCell) {
            // Try to convert whatever it is to date
            const dateObj = new Date(remedialDateCell);
            remedialDateStr = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
          }
        } catch (e) {
          Logger.log(`  Error converting date: ${e.message}`);
          continue;
        }
        
        Logger.log(`  Date (formatted): "${remedialDateStr}"`);
        Logger.log(`  Subject: "${row[2]}"`);
        Logger.log(`  Class: "${row[3]}"`);
        
        // FIXED: Compare trimmed teacher names and check for 'Scheduled' status
        if (rowTeacher.trim() === teacherName.trim() && rowStatus === 'Scheduled') {
          Logger.log(`  ✓ Found scheduled lesson at row ${i+1}`);
          
          // Check if remedial date is today or in the future
          if (remedialDateStr && remedialDateStr >= todayStr) {
            scheduledLessons.push({
              originalDate: formatDateForDisplay(row[0]),
              subject: row[2] ? row[2].toString().trim() : '',
              className: row[3] ? row[3].toString().trim() : '',
              remedialDate: remedialDateStr,
              remedialTime: row[5] ? row[5].toString().trim() : '',
              duration: row[6] ? row[6].toString().trim() : '',
              topic: row[7] ? row[7].toString().trim() : '',
              location: row[8] ? row[8].toString().trim() : '',
              status: rowStatus,
              rowIndex: i + 1
            });
            Logger.log(`  ✓ Added to list (date ${remedialDateStr} >= ${todayStr})`);
          } else {
            Logger.log(`  ✗ Skipped (date ${remedialDateStr} is before today ${todayStr} or invalid)`);
          }
        }
      }
    }
    
    Logger.log(`\nTotal scheduled lessons found: ${scheduledLessons.length}`);
    
    // Sort by date (soonest first)
    scheduledLessons.sort((a, b) => {
      const dateA = new Date(a.remedialDate);
      const dateB = new Date(b.remedialDate);
      return dateA - dateB;
    });
    
    // Log final results
    if (scheduledLessons.length > 0) {
      Logger.log('Final scheduled lessons:');
      scheduledLessons.forEach((lesson, index) => {
        Logger.log(`  ${index + 1}. ${lesson.subject} - ${lesson.className} on ${lesson.remedialDate} at ${lesson.remedialTime}`);
      });
    } else {
      Logger.log('No scheduled lessons found for this teacher');
    }
    
    Logger.log('=== GET SCHEDULED REMEDIAL LESSONS - COMPLETE ===');
    return scheduledLessons;
    
  } catch (error) {
    Logger.log('=== GET SCHEDULED REMEDIAL LESSONS - ERROR ===');
    Logger.log('Error: ' + error.message);
    Logger.log('Stack: ' + error.stack);
    return [];
  }
}

// Helper function to format dates for display
function formatDateForDisplay(dateValue) {
  try {
    if (dateValue instanceof Date) {
      return Utilities.formatDate(dateValue, 'Africa/Accra', 'yyyy-MM-dd');
    } else if (typeof dateValue === 'string') {
      const dateObj = new Date(dateValue);
      return Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
    }
    return dateValue ? dateValue.toString() : '';
  } catch (e) {
    return dateValue ? dateValue.toString() : '';
  }
}

// Helper function to normalize dates for comparison
function normalizeDateForComparison(dateValue) {
  try {
    if (!dateValue) return '';
    
    if (dateValue instanceof Date) {
      return Utilities.formatDate(dateValue, 'Africa/Accra', 'yyyy-MM-dd');
    }
    
    if (typeof dateValue === 'string') {
      // Remove any time portion
      const dateStr = dateValue.split(' ')[0];
      
      // Check if already in YYYY-MM-DD format
      if (dateStr.match(/^\d{4}-\d{2}-\d{2}$/)) {
        return dateStr;
      }
      
      // Try to parse as date
      const dateObj = new Date(dateStr);
      if (!isNaN(dateObj.getTime())) {
        return Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
      }
      
      return dateStr.trim();
    }
    
    return dateValue.toString().trim();
  } catch (e) {
    Logger.log('Error normalizing date: ' + e.message);
    return dateValue ? dateValue.toString().trim() : '';
  }
}

/**
 * Extract photo URL from cell value (handles both formulas and plain text)
 */
function extractPhotoUrl(cellValue) {
  try {
    if (!cellValue) return '';
    
    const cellStr = cellValue.toString().trim();
    
    // If it's a hyperlink formula
    if (cellStr.includes('=HYPERLINK(')) {
      const urlMatch = cellStr.match(/=HYPERLINK\("([^"]+)"(?:,\s*"([^"]+)")?\)/);
      if (urlMatch && urlMatch[1]) {
        return urlMatch[1];
      }
    }
    
    // If it's already a URL
    if (cellStr.startsWith('http')) {
      return cellStr;
    }
    
    // Check if it's a Drive file ID
    if (cellStr.startsWith('https://drive.google.com/file/d/')) {
      return cellStr;
    }
    
    return cellStr;
  } catch (error) {
    console.log('Error extracting photo URL:', error);
    return cellValue ? cellValue.toString().trim() : '';
  }
}

          rowOriginalDateStr = Utilities.formatDate(rowOriginalDate, 'Africa/Accra', 'yyyy-MM-dd');
        } else if (typeof rowOriginalDate === 'string') {
          const dateObj = new Date(rowOriginalDate);
          rowOriginalDateStr = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
        } else {
          rowOriginalDateStr = rowOriginalDate ? rowOriginalDate.toString().trim() : '';
        }
      } catch (e) {
        rowOriginalDateStr = rowOriginalDate ? rowOriginalDate.toString().trim() : '';
      }
      
      // Debug logging
      Logger.log(`Checking row ${i + 1}: ${rowTeacher} - ${rowSubject} - ${rowClassName}`);
      Logger.log(`  Status: "${rowStatus}", Looking for: "Completed"`);
      Logger.log(`  Date match? "${rowOriginalDateStr}" === "${originalDate}"`);
      
      if (rowTeacher === teacherName && 
          rowOriginalDateStr === originalDate && 
          rowSubject === subject && 
          rowClassName === className &&
          rowStatus === 'Completed') {
        
        targetRow = i + 1;
        targetRemedialDate = rowRemedialDate;
        Logger.log(`✓ Found completed remedial lesson at row ${targetRow}`);
        break;
      }
    }
    
    if (!targetRow) {
      // More helpful error message
      Logger.log(`❌ No completed remedial lesson found for:`);
      Logger.log(`  Teacher: ${teacherName}`);
      Logger.log(`  Date: ${originalDate}`);
      Logger.log(`  Subject: ${subject}`);
      Logger.log(`  Class: ${className}`);
      throw new Error('No completed remedial lesson found for verification. Status must be "Completed".');
    }
    
    // Mark as verified
    const now = new Date();
    const timeZone = 'Africa/Accra';
    const verifier = Session.getEffectiveUser().getEmail() || 'Admin';
    const currentDate = Utilities.formatDate(now, timeZone, 'yyyy-MM-dd');
    const currentTime = Utilities.formatDate(now, timeZone, 'HH:mm:ss');
    const currentDateTime = Utilities.formatDate(now, timeZone, 'yyyy-MM-dd HH:mm:ss');
    
    // Update columns: L=12, M=13, N=14, O=15, R=18
    remedialSheet.getRange(targetRow, 12).setValue('Verified'); // Status
    remedialSheet.getRange(targetRow, 13).setValue(verifier);   // Verified By
    remedialSheet.getRange(targetRow, 14).setValue(currentDate); // Verified Date
    remedialSheet.getRange(targetRow, 15).setValue(
      (remedialSheet.getRange(targetRow, 15).getValue() || '') + 
      `\nVerified by admin at ${currentTime}`
    ); // Notes
    remedialSheet.getRange(targetRow, 18).setValue(currentDateTime); // Last Updated
    
    // Find and remove corresponding absence record
    const absenceData = absenceSheet.getDataRange().getValues();
    const absencesToDelete = [];
    
    for (let i = 1; i < absenceData.length; i++) {
      const row = absenceData[i];
      const absenceTeacher = row[2] ? row[2].toString().trim() : '';
      const absenceDate = row[0];
      const absenceSubject = row[3] ? row[3].toString().trim() : '';
      const absenceClass = row[4] ? row[4].toString().trim() : '';
      const autoGenerated = row[7] ? row[7].toString().trim() : '';
      
      if (absenceTeacher === teacherName && 
          autoGenerated === 'Yes' && 
          absenceSubject === subject && 
          absenceClass === className) {
        
        // Convert dates for comparison
        let absenceDateStr;
        try {
          if (absenceDate instanceof Date) {
            absenceDateStr = Utilities.formatDate(absenceDate, timeZone, 'yyyy-MM-dd');
          } else {
            const dateObj = new Date(absenceDate);
            absenceDateStr = Utilities.formatDate(dateObj, timeZone, 'yyyy-MM-dd');
          }
        } catch (e) {
          continue;
        }
        
        if (absenceDateStr === originalDate) {
          absencesToDelete.push(i + 1);
          Logger.log(`Found matching absence at row ${i + 1}: ${absenceDateStr}`);
        }
      }
    }
    
    // Delete absence records from bottom to top
    let deletedCount = 0;
    absencesToDelete.sort((a, b) => b - a).forEach(rowIndex => {
      absenceSheet.deleteRow(rowIndex);
      deletedCount++;
      Logger.log(`Deleted absence record at row ${rowIndex}`);
    });
    
    // Update the Absences sheet if the original absence record was not auto-generated
    // This is for absences that were manually recorded
    if (deletedCount === 0) {
      Logger.log('No auto-generated absences found to delete.');
      Logger.log('Checking for manual absence records to update...');
      
      for (let i = 1; i < absenceData.length; i++) {
        const row = absenceData[i];
        const absenceTeacher = row[2] ? row[2].toString().trim() : '';
        const absenceDate = row[0];
        const absenceSubject = row[3] ? row[3].toString().trim() : '';
        const absenceClass = row[4] ? row[4].toString().trim() : '';
        const absenceStatus = row[6] ? row[6].toString().trim() : '';
        
        if (absenceTeacher === teacherName && 
            absenceSubject === subject && 
            absenceClass === className &&
            (absenceStatus === 'Absent' || absenceStatus.includes('Absent'))) {
          
          // Convert dates for comparison
          let absenceDateStr;
          try {
            if (absenceDate instanceof Date) {
              absenceDateStr = Utilities.formatDate(absenceDate, timeZone, 'yyyy-MM-dd');
            } else {
              const dateObj = new Date(absenceDate);
              absenceDateStr = Utilities.formatDate(dateObj, timeZone, 'yyyy-MM-dd');
            }
          } catch (e) {
            continue;
          }
          
          if (absenceDateStr === originalDate) {
            // Update status to show remedial was verified
            absenceSheet.getRange(i + 1, 7).setValue('Remedial Verified & Cleared'); // Column G: Status
            absenceSheet.getRange(i + 1, 9).setValue(
              (row[8] || '') + ` | Remedial verified on ${currentDate} by ${verifier}`
            ); // Column I: Reason/Notes
            deletedCount++;
            Logger.log(`Updated manual absence at row ${i + 1}`);
          }
        }
      }
    }
    
    // Send notification to teacher
    sendRemedialVerifiedNotification(teacherName, subject, originalDate, targetRemedialDate);
    
    Logger.log(`Verified remedial lesson for ${teacherName}, cleared/updated ${deletedCount} absence record(s)`);
    
    return {
      success: true,
      message: `Remedial lesson verified successfully. ${deletedCount} absence record(s) cleared/updated.`,
      clearedAbsences: deletedCount,
      teacher: teacherName,
      subject: subject,
      originalDate: originalDate,
      remedialDate: targetRemedialDate
    };
    
  } catch (error) {
    Logger.log('Error verifying remedial lesson: ' + error.message);
    Logger.log('Stack: ' + error.stack);
    return {
      success: false,
      message: '❌ ' + error.message,
      clearedAbsences: 0
    };
  }
}

// Generate statistics about remedial lessons
function generateRemedialLessonReport(startDate = '', endDate = '') {
  try {
    initializeRemedialSheet();
    
    const remedialSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('RemedialLessons');
    const absenceSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Absences');
    
    if (!remedialSheet || !absenceSheet) {
      return {
        success: false,
        message: 'Required sheets not found'
      };
    }
    
    const remedialData = remedialSheet.getDataRange().getValues();
    const absenceData = absenceSheet.getDataRange().getValues();
    
    const stats = {
      totalAbsences: 0,
      scheduledRemedials: 0,
      completedRemedials: 0,
      verifiedRemedials: 0,
      coverageRate: 0,
      byTeacher: {}
    };
    
    // Calculate date range if provided
    const start = startDate ? new Date(startDate) : null;
    const end = endDate ? new Date(endDate) : null;
    
    // Count auto-generated absences
    for (let i = 1; i < absenceData.length; i++) {
      const row = absenceData[i];
      const autoGenerated = row[7];
      const teacher = row[2];
      
      if (autoGenerated === 'Yes') {
        // Check date range
        const absenceDate = row[0];
        let absenceDateObj;
        try {
          if (absenceDate instanceof Date) {
            absenceDateObj = absenceDate;
          } else {
            absenceDateObj = new Date(absenceDate);
          }
        } catch (e) {
          continue;
        }
        
        // Apply date filter
        if ((!start || absenceDateObj >= start) && (!end || absenceDateObj <= end)) {
          stats.totalAbsences++;
          
          // Initialize teacher stats
          if (!stats.byTeacher[teacher]) {
            stats.byTeacher[teacher] = {
              totalAbsences: 0,
              scheduledRemedials: 0,
              completedRemedials: 0,
              verifiedRemedials: 0
            };
          }
          stats.byTeacher[teacher].totalAbsences++;
        }
      }
    }
    
    // Count remedial lessons
    for (let i = 1; i < remedialData.length; i++) {
      const row = remedialData[i];
      const teacher = row[1];
      const status = row[11];
      const remedialDate = row[4];
      
      // Check date range for remedial date
      if (remedialDate) {
        let remedialDateObj;
        try {
          if (remedialDate instanceof Date) {
            remedialDateObj = remedialDate;
          } else {
            remedialDateObj = new Date(remedialDate);
          }
        } catch (e) {
          continue;
        }
        
        // Apply date filter
        if ((!start || remedialDateObj >= start) && (!end || remedialDateObj <= end)) {
          // Initialize teacher stats if not exists
          if (!stats.byTeacher[teacher]) {
            stats.byTeacher[teacher] = {
              totalAbsences: 0,
              scheduledRemedials: 0,
              completedRemedials: 0,
              verifiedRemedials: 0
            };
          }
          
          // Count by status
          switch(status) {
            case 'Scheduled':
              stats.scheduledRemedials++;
              stats.byTeacher[teacher].scheduledRemedials++;
              break;
            case 'Completed':
              stats.completedRemedials++;
              stats.byTeacher[teacher].completedRemedials++;
              break;
            case 'Verified':
              stats.verifiedRemedials++;
              stats.byTeacher[teacher].verifiedRemedials++;
              break;
          }
        }
      }
    }
    
    // Calculate coverage rate
    if (stats.totalAbsences > 0) {
      stats.coverageRate = Math.round((stats.verifiedRemedials / stats.totalAbsences) * 100);
    }
    
    return stats;
    
  } catch (error) {
    Logger.log('Error generating remedial report: ' + error.message);
    return {
      success: false,
      message: error.message
    };
  }
}

/**
 * Manually record absences for a specific date
 * @param {string} dateStr - Date in YYYY-MM-DD format
 * @param {string} teacherFilter - Optional teacher name filter
 * @returns {Object} Result with details
 */
function manualRecordAbsences(dateStr, teacherFilter = '') {
  try {
    Logger.log(`=== MANUAL ABSENCE RECORDING FOR: ${dateStr} ===`);
    
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const timetableSheet = ss.getSheetByName('Timetable');
    const formDataSheet = ss.getSheetByName('FormData');
    const absenceSheet = ss.getSheetByName('Absences');
    
    if (!timetableSheet || !formDataSheet) {
      throw new Error('Required sheets not found');
    }
    
    // Ensure Absences sheet exists
    if (!absenceSheet) {
      const newAbsenceSheet = ss.insertSheet('Absences');
      newAbsenceSheet.getRange('A1:I1').setValues([[
        'Date', 'Time', 'Teacher', 'Subject', 'Class',
        'Scheduled Period', 'Status', 'Auto Generated', 'Reason'
      ]]);
      Logger.log('Created new Absences sheet');
    }
    
    const timetableData = timetableSheet.getDataRange().getValues();
    const attendanceData = formDataSheet.getDataRange().getValues();
    
    // Validate date
    const targetDate = new Date(dateStr);
    if (isNaN(targetDate.getTime())) {
      throw new Error('Invalid date format. Please use YYYY-MM-DD');
    }
    
    const dayOfWeek = getDayOfWeekFromDate(targetDate);
    const dayName = getDayName(dayOfWeek === 7 ? 0 : dayOfWeek);
    
    Logger.log(`Processing date: ${dateStr} (${dayName})`);
    
    // Find absences for this date
    const absences = findAbsencesForManualDate(dateStr, timetableData, attendanceData, teacherFilter);
    
    // Record absences
    const recordedCount = recordManualAbsencesForDate(dateStr, absences);
    
    Logger.log(`✅ Manual recording completed: ${recordedCount} absences recorded`);
    
    return {
      success: true,
      date: dateStr,
      day: dayName,
      totalLessons: absences.length,
      absencesRecorded: recordedCount,
      message: `Successfully recorded ${recordedCount} absences for ${dateStr} (${dayName})`,
      absences: absences.slice(0, 10) // Return first 10 for preview
    };
    
  } catch (error) {
    Logger.log('❌ ERROR in manualRecordAbsences: ' + error.toString());
    return {
      success: false,
      date: dateStr,
      totalLessons: 0,
      absencesRecorded: 0,
      message: 'Failed to record absences: ' + error.message,
      absences: []
    };
  }
}

/**
 * Find absences for a manual date with optional teacher filter
 */
function findAbsencesForManualDate(dateStr, timetableData, attendanceData, teacherFilter = '') {
  const absences = [];
  
  try {
    const dateObj = new Date(dateStr);
    const dayOfWeek = getDayOfWeekFromDate(dateObj);
    
    Logger.log(`  Looking for timetable entries for ${getDayName(dayOfWeek)} (Day ${dayOfWeek})`);
    
    // Find all timetable entries for this day of week
    const dailyTimetable = [];
    for (let i = 1; i < timetableData.length; i++) {
      const row = timetableData[i];
      const timetableDay = parseInt(row[0], 10);
      
      if (timetableDay === dayOfWeek) {
        const teacher = row[3] ? row[3].toString().trim() : '';
        const subject = row[4] ? row[4].toString().trim() : '';
        const className = row[5] ? row[5].toString().trim() : '';
        const startTime = formatTimeForCalculation(row[1]);
        const endTime = formatTimeForCalculation(row[2]);
        
        if (teacher && subject && className) {
          // Apply teacher filter if specified
          if (teacherFilter && teacher !== teacherFilter) {
            continue;
          }
          
          dailyTimetable.push({
            teacher: teacher,
            subject: subject,
            className: className,
            startTime: startTime,
            endTime: endTime,
            period: `${startTime}-${endTime}`
          });
        }
      }
    }
    
    Logger.log(`  Found ${dailyTimetable.length} scheduled lessons`);
    
    // Check each scheduled lesson for attendance
    dailyTimetable.forEach(lesson => {
      const hasAttendance = checkIfAttendanceExistsForManualDate(attendanceData, dateStr, lesson);
      
      if (!hasAttendance) {
        absences.push({
          date: dateStr,
          teacher: lesson.teacher,
          subject: lesson.subject,
          className: lesson.className,
          period: lesson.period,
          startTime: lesson.startTime,
          endTime: lesson.endTime,
          status: 'Absent'
        });
        
        Logger.log(`  ❌ ABSENCE: ${lesson.teacher} - ${lesson.subject} - ${lesson.className}`);
      } else {
        Logger.log(`  ✅ PRESENT: ${lesson.teacher} - ${lesson.subject} - ${lesson.className}`);
      }
    });
    
  } catch (error) {
    Logger.log(`  ERROR processing date ${dateStr}: ` + error.toString());
  }
  
  return absences;
}

/**
 * Check if attendance exists for a manual date
 */
function checkIfAttendanceExistsForManualDate(attendanceData, dateStr, lesson) {
  for (let i = 1; i < attendanceData.length; i++) {
    const row = attendanceData[i];
    if (!row[0]) continue;
    
    // Parse record date
    let recordDate = '';
    try {
      if (row[0] instanceof Date) {
        recordDate = Utilities.formatDate(row[0], 'Africa/Accra', 'yyyy-MM-dd');
      } else if (typeof row[0] === 'string') {
        const dateObj = new Date(row[0]);
        recordDate = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
      }
    } catch (e) {
      continue;
    }
    
    const teacher = row[4] ? row[4].toString().trim() : '';
    const subject = row[5] ? row[5].toString().trim() : '';
    const classes = row[6] ? row[6].toString().trim() : '';
    
    // Check for match
    if (recordDate === dateStr && 
        teacher === lesson.teacher && 
        subject === lesson.subject &&
        classes.includes(lesson.className)) {
      return true;
    }
  }
  return false;
}

/**
 * Record manual absences for a date
 */
function recordManualAbsencesForDate(dateStr, absences) {
  if (absences.length === 0) return 0;
  
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const absenceSheet = ss.getSheetByName('Absences');
  
  let recordedCount = 0;
  
  absences.forEach(absence => {
    // Check if absence is already recorded
    if (!isAbsenceAlreadyRecordedManual(absenceSheet, absence)) {
      // Record the absence
      const now = new Date();
      const recordTime = Utilities.formatDate(now, 'Africa/Accra', 'HH:mm');
      
      absenceSheet.appendRow([
        absence.date,
        recordTime,
        absence.teacher,
        absence.subject,
        absence.className,
        absence.period,
        'Absent',
        'Yes (Manual Recording)',
        'Recorded manually by administrator'
      ]);
      
      recordedCount++;
      Logger.log(`      📝 Recorded: ${absence.teacher} - ${absence.subject}`);
    } else {
      Logger.log(`      ⏭️  Already recorded: ${absence.teacher} - ${absence.subject}`);
    }
  });
  
  return recordedCount;
}

/**
 * Check if absence is already recorded (manual version)
 */
function isAbsenceAlreadyRecordedManual(absenceSheet, absence) {
  const data = absenceSheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const recordDate = row[0];
    const teacher = row[2];
    const subject = row[3];
    const className = row[4];
    
    let recordDateStr = '';
    try {
      if (recordDate instanceof Date) {
        recordDateStr = Utilities.formatDate(recordDate, 'Africa/Accra', 'yyyy-MM-dd');
      } else if (typeof recordDate === 'string') {
        const dateObj = new Date(recordDate);
        recordDateStr = Utilities.formatDate(dateObj, 'Africa/Accra', 'yyyy-MM-dd');
      }
    } catch (e) {
      continue;
    }
    
    if (recordDateStr === absence.date && 
        teacher === absence.teacher && 
        subject === absence.subject &&
        className === absence.className) {
      return true;
    }
  }
  return false;
}

/**
 * Preview absences without recording (safe mode)
 */
function previewManualAbsences(dateStr, teacherFilter = '') {
  try {
    Logger.log(`=== PREVIEWING ABSENCES FOR: ${dateStr} ===`);
    
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const timetableSheet = ss.getSheetByName('Timetable');
    const formDataSheet = ss.getSheetByName('FormData');
    
    if (!timetableSheet || !formDataSheet) {
      throw new Error('Required sheets not found');
    }
    
    const timetableData = timetableSheet.getDataRange().getValues();
    const attendanceData = formDataSheet.getDataRange().getValues();
    
    // Validate date
    const targetDate = new Date(dateStr);
    if (isNaN(targetDate.getTime())) {
      throw new Error('Invalid date format. Please use YYYY-MM-DD');
    }
    
    const dayOfWeek = getDayOfWeekFromDate(targetDate);
    const dayName = getDayName(dayOfWeek === 7 ? 0 : dayOfWeek);
    
    // Find absences for this date
    const absences = findAbsencesForManualDate(dateStr, timetableData, attendanceData, teacherFilter);
    
    Logger.log(`✅ Preview completed: ${absences.length} potential absences found`);
    
    return {
      success: true,
      date: dateStr,
      day: dayName,
      totalLessons: absences.length,
      absencesFound: absences.length,
      message: `Found ${absences.length} potential absences for ${dateStr} (${dayName})`,
      absences: absences
    };
    
  } catch (error) {
    Logger.log('❌ ERROR in previewManualAbsences: ' + error.toString());
    return {
      success: false,
      date: dateStr,
      totalLessons: 0,
      absencesFound: 0,
      message: 'Failed to preview absences: ' + error.message,
      absences: []
    };
  }
}

/**
 * Record absences for a week (Monday-Friday)
 */
function manualRecordAbsencesForWeek(startDateStr) {
  try {
    Logger.log(`=== RECORDING ABSENCES FOR WEEK STARTING: ${startDateStr} ===`);
    
    const startDate = new Date(startDateStr);
    const endDate = new Date(startDate);
    endDate.setDate(startDate.getDate() + 4); // Monday to Friday (5 days)
    
    let totalAbsencesRecorded = 0;
    let totalDatesProcessed = 0;
    const results = [];
    
    // Process each day from Monday to Friday
    let currentDate = new Date(startDate);
    while (currentDate <= endDate) {
      const dateStr = Utilities.formatDate(currentDate, 'Africa/Accra', 'yyyy-MM-dd');
      const dayOfWeek = currentDate.getDay();
      
      // Only process weekdays (Monday=1 to Friday=5)
      if (dayOfWeek >= 1 && dayOfWeek <= 5) {
        totalDatesProcessed++;
        
        const result = manualRecordAbsences(dateStr);
        if (result.success) {
          totalAbsencesRecorded += result.absencesRecorded;
          results.push({
            date: dateStr,
            absences: result.absencesRecorded,
            message: result.message
          });
        }
      }
      
      currentDate.setDate(currentDate.getDate() + 1);
    }
    
    Logger.log(`✅ Week recording completed: ${totalAbsencesRecorded} absences recorded across ${totalDatesProcessed} days`);
    
    return {
      success: true,
      weekStart: startDateStr,
      daysProcessed: totalDatesProcessed,
      absencesRecorded: totalAbsencesRecorded,
      message: `Recorded ${totalAbsencesRecorded} absences across ${totalDatesProcessed} days`,
      dailyResults: results
    };
    
  } catch (error) {
    Logger.log('❌ ERROR in manualRecordAbsencesForWeek: ' + error.toString());
    return {
      success: false,
      weekStart: startDateStr,
      daysProcessed: 0,
      absencesRecorded: 0,
      message: 'Failed to record weekly absences: ' + error.message,
      dailyResults: []
    };
  }
}

/**
 * Get last Friday's date
 */
function getLastFriday() {
  const today = new Date();
  const dayOfWeek = today.getDay();
  
  // If today is Friday, return today, otherwise go back to previous Friday
  const daysSinceFriday = dayOfWeek === 5 ? 0 : (dayOfWeek + 2) % 7;
  const lastFriday = new Date(today);
  lastFriday.setDate(today.getDate() - daysSinceFriday);
  
  return Utilities.formatDate(lastFriday, 'Africa/Accra', 'yyyy-MM-dd');
}

/**
 * Get last Monday's date
 */
function getLastMonday() {
  const today = new Date();
  const dayOfWeek = today.getDay();
  
  // If today is Monday, return today, otherwise go back to previous Monday
  const daysSinceMonday = dayOfWeek === 1 ? 0 : (dayOfWeek + 6) % 7;
  const lastMonday = new Date(today);
  lastMonday.setDate(today.getDate() - daysSinceMonday);
  
  return Utilities.formatDate(lastMonday, 'Africa/Accra', 'yyyy-MM-dd');
}
