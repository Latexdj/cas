    // Automatically get GPS coordinates
    function getGPS() {
      const gpsStatus = document.getElementById('gpsStatus');
      
      if (navigator.geolocation) {
        const options = {
          enableHighAccuracy: true,
          timeout: 75000,
          maximumAge: 0
        };
        
        navigator.geolocation.getCurrentPosition(
          function(position) {
            const lat = position.coords.latitude;
            const lng = position.coords.longitude;
            const accuracy = position.coords.accuracy;
            
            document.getElementById('gps').value = lat + "," + lng;
            gpsStatus.innerHTML = '<i class="fas fa-check-circle success"></i> Location captured (Accuracy: ' + Math.round(accuracy) + 'm)';
            gpsStatus.className = 'gps-status success';
            
            // Trigger location verification if a location is selected
            updateLocationVerification();
          }, 
          function(error) {
            console.log("Error getting location: " + error.message);
            let errorMsg = 'Error getting location: ';
            switch(error.code) {
              case error.PERMISSION_DENIED:
                errorMsg += 'Permission denied. Please enable location services.';
                break;
              case error.POSITION_UNAVAILABLE:
                errorMsg += 'Location unavailable. Please check your mobile data.';
                break;
              case error.TIMEOUT:
                errorMsg += 'Location request timed out. Please try again.';
                break;
              default:
                errorMsg += error.message;
            }
            gpsStatus.innerHTML = '<i class="fas fa-exclamation-circle error"></i> ' + errorMsg;
            gpsStatus.className = 'gps-status error';
          },
          options
        );
      } else {
        gpsStatus.innerHTML = '<i class="fas fa-exclamation-circle error"></i> Geolocation is not supported by this browser';
        gpsStatus.className = 'gps-status error';
      }
    }

    // Update location verification when location changes
    function updateLocationVerification() {
      const locationSelect = document.getElementById('teachingLocation');
      const locationVerification = document.getElementById('locationVerification');
      const verificationResult = document.getElementById('verificationResult');
      const gpsValue = document.getElementById('gps').value;
      
      if (locationSelect.value && gpsValue && gpsValue.includes(',')) {
        const coords = gpsValue.split(',');
        const lat = parseFloat(coords[0]);
        const lng = parseFloat(coords[1]);
        
        locationVerification.style.display = 'block';
        verificationResult.innerHTML = '<i class="fas fa-sync-alt fa-spin"></i> Verifying location...';
        
        google.script.run
          .withSuccessHandler(function(result) {
            locationVerification.className = 'location-verification';
            
            if (result.verified) {
              if (result.valid) {
                locationVerification.classList.add('verified');
                verificationResult.innerHTML = '<span class="success"><i class="fas fa-check-circle"></i> ' + result.message + '</span>';
              } else {
                locationVerification.classList.add('error');
                verificationResult.innerHTML = '<span class="error"><i class="fas fa-exclamation-circle"></i> ' + result.message + '</span>';
              }
            } else {
              locationVerification.classList.add('not-verified');
              verificationResult.innerHTML = '<span class="warning"><i class="fas fa-info-circle"></i> ' + result.message + '</span>';
            }
          })
          .withFailureHandler(function(error) {
            locationVerification.className = 'location-verification error';
            verificationResult.innerHTML = '<span class="error"><i class="fas fa-exclamation-triangle"></i> Verification error: ' + error + '</span>';
          })
          .verifyTeachingLocation(locationSelect.value, lat, lng);
      } else if (locationSelect.value) {
        locationVerification.style.display = 'block';
        locationVerification.className = 'location-verification not-verified';
        verificationResult.innerHTML = '<span class="warning"><i class="fas fa-info-circle"></i> Enable location services to verify your position</span>';
      } else {
        locationVerification.style.display = 'none';
      }
    }

    // Update subject dropdown based on selected class
    function updateSubjectDropdown() {
      const selectedClasses = document.querySelectorAll('input[name="className"]:checked');
      const subjectSelect = document.getElementById('subject');
      
      subjectSelect.innerHTML = '<option value="">-- Select Subject --</option>';
      
      if (selectedClasses.length === 0) {
        return;
      }
      
      // For single class selection, auto-select the subject
      if (selectedClasses.length === 1) {
        const subject = selectedClasses[0].dataset.subject;
        const option = document.createElement('option');
        option.value = subject;
        option.textContent = subject;
        option.selected = true;
        subjectSelect.appendChild(option);
        selectedClassDetails.subject = subject;
        selectedClassDetails.periods = selectedClasses[0].dataset.periods;
      } else {
        // For multiple classes, show all subjects
        const subjects = new Set();
        selectedClasses.forEach(checkbox => {
          subjects.add(checkbox.dataset.subject);
        });
        
        subjects.forEach(subject => {
          const option = document.createElement('option');
          option.value = subject;
          option.textContent = subject;
          subjectSelect.appendChild(option);
        });
      }
    }

    // Update periods dropdown based on selected class and subject
    function updatePeriodsForSelectedClass() {
      const selectedClasses = document.querySelectorAll('input[name="className"]:checked');
      const selectedSubject = document.getElementById('subject').value;
      const periodsSelect = document.getElementById('periods');
      
      periodsSelect.innerHTML = '<option value="">-- Select Periods --</option>';
      
      if (selectedClasses.length === 0 || !selectedSubject) {
        return;
      }
      
      // Find the class that matches the selected subject
      let periods = null;
      selectedClasses.forEach(checkbox => {
        if (checkbox.dataset.subject === selectedSubject) {
          periods = checkbox.dataset.periods;
        }
      });
      
      if (periods) {
        const option = document.createElement('option');
        option.value = periods;
        option.textContent = periods + ' period(s)';
        option.selected = true;
        periodsSelect.appendChild(option);
      }
    }

    function submitForm() {
      // Prevent multiple submissions using global flag
      if (isSubmitting) {
        alert('Please wait... Your attendance is being submitted.');
        return;
      }
      
      // Set submission flag immediately
      isSubmitting = true;
      
      // Get the submit button and disable it immediately
      const submitButton = document.querySelector('.btn-submit');
      const originalButtonText = submitButton.innerHTML;
      
      // Disable the button and show loading state
      submitButton.disabled = true;
      submitButton.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Submitting...';
      submitButton.style.opacity = '0.7';
      submitButton.style.cursor = 'not-allowed';
      
      const form = document.getElementById('myForm');
      const academicYear = document.getElementById('academicYear').value;
      const semester = document.getElementById('semester').value;
      const name = document.getElementById('name').value;
      const subject = document.getElementById('subject').value;
      const periods = document.getElementById('periods').value;
      const topic = document.getElementById('topic').value;
      const gps = document.getElementById('gps').value;
      const teachingLocation = document.getElementById('teachingLocation').value;
      const imageFile = document.getElementById('photo').files[0];

      // Collect selected class names
      const classCheckboxes = document.querySelectorAll('input[name="className"]:checked');
      const classNames = Array.from(classCheckboxes).map(checkbox => checkbox.value).join(', ');

      // Validate form fields
      if (!academicYear || !semester || !name || !subject || !classNames || !periods || !topic || !gps || !teachingLocation) {
        alert('Please fill in all required fields.');
        resetSubmitState(submitButton, originalButtonText);
        return;
      }

      if (!imageFile) {
        alert('Please capture a photo.');
        resetSubmitState(submitButton, originalButtonText);
        return;
      }

      // Show progress bar
      document.getElementById('progressBar').style.width = '0%';
      document.getElementById('progressBar').textContent = '0%';

      // Convert image to base64
      const reader = new FileReader();
      
      reader.onload = function () {
        const imageBase64 = reader.result;

        // Create data object
        const data = {
          academicYear: academicYear,
          semester: semester,
          name: name,
          subject: subject,
          classNames: classNames,
          periods: periods,
          topic: topic,
          gps: gps,
          teachingLocation: teachingLocation,
          imageBase64: imageBase64
        };

        // Simulate progress
        let width = 1;
        const interval = setInterval(function() {
          if (width >= 100) {
            clearInterval(interval);
          } else {
            width++;
            document.getElementById('progressBar').style.width = width + '%';
            document.getElementById('progressBar').textContent = width + '%';
          }
        }, 150);

        // Send data to Google Apps Script
        google.script.run
          .withSuccessHandler(function(response) {
            clearInterval(interval);
            document.getElementById('progressBar').style.width = '100%';
            document.getElementById('progressBar').textContent = '100%';
            
            setTimeout(function() {
              alert(response);
              form.reset();
              getGPS();
              document.getElementById('absenceNotice').style.display = 'none';
              document.getElementById('absenceReasonSection').style.display = 'none';
              
              // Reload the data to maintain auto-selection
              loadAcademicYearsAndSemesters();
              loadTeachingLocations();
              
              // Reset submission state after successful submission
              resetSubmitState(submitButton, originalButtonText);
            }, 1000);
          })
          .withFailureHandler(function(error) {
            clearInterval(interval);
            document.getElementById('progressBar').style.width = '100%';
            document.getElementById('progressBar').textContent = '100%';
            
            alert('Error: ' + error.message);
            
            // Reset submission state on error
            resetSubmitState(submitButton, originalButtonText);
          })
          .submitData(data);
      };
      
      reader.onerror = function() {
        alert('Error reading the photo. Please try again.');
        resetSubmitState(submitButton, originalButtonText);
      };
      
      reader.onabort = function() {
        alert('Photo reading was cancelled. Please try again.');
        resetSubmitState(submitButton, originalButtonText);
      };
      
      reader.readAsDataURL(imageFile);
    }

      button.innerHTML = originalText;
      button.style.opacity = '1';
      button.style.cursor = 'pointer';
    }

// Helper function to escape HTML special characters
function escapeHtml(text) {
  if (!text) return '';
  return text.toString()
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#039;');
}

    // Get GPS for remedial lesson
    function getGPSForRemedial() {
      const remedialGPSInput = document.getElementById('remedialGPS');
      if (!remedialGPSInput) {
        console.error('remedialGPS input not found');
        return;
      }
      
      remedialGPSInput.value = 'Getting location...';
      
      if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(
          function(position) {
            const lat = position.coords.latitude;
            const lng = position.coords.longitude;
            const accuracy = position.coords.accuracy;
            remedialGPSInput.value = `${lat},${lng}`;
            console.log(`GPS captured for remedial: ${lat},${lng} (Accuracy: ${accuracy}m)`);
          },
          function(error) {
            remedialGPSInput.value = 'Error getting location. Please enable GPS.';
            console.error('GPS error for remedial:', error);
          },
          { enableHighAccuracy: true, timeout: 10000 }
        );
      } else {
        remedialGPSInput.value = 'GPS not supported by browser';
      }
    }

    // Submit remedial attendance
    function submitRemedialAttendance() {
      const teacherName = document.getElementById('name').value;
      const remedialGPSInput = document.getElementById('remedialGPS');
      const photoInput = document.getElementById('remedialPhoto');
      
      // Get stored lesson
      const lessonData = sessionStorage.getItem('selectedRemedialLesson');
      if (!lessonData) {
        alert('No lesson selected. Please select a lesson first.');
        return;
      }
      
      const lesson = JSON.parse(lessonData);
      
      // DEBUG: Show what we're sending
      console.log('DEBUG - Submitting remedial attendance with:');
      console.log('Teacher:', teacherName);
      console.log('Original Date:', lesson.originalDate);
      console.log('Subject:', lesson.subject);
      console.log('ClassName:', lesson.className);
      console.log('Remedial Date:', lesson.remedialDate);
      console.log('Remedial Time:', lesson.remedialTime);
      
      if (!remedialGPSInput || !remedialGPSInput.value || remedialGPSInput.value === 'Getting location...') {
        alert('Please get your GPS location first');
        return;
      }
      
      if (!photoInput || !photoInput.files[0]) {
        alert('Please take a photo of the remedial lesson');
        return;
      }
      
      // Convert photo to base64
      const reader = new FileReader();
      reader.onload = function() {
        const data = {
          teacher: teacherName,
          originalDate: lesson.originalDate, // This should be in YYYY-MM-DD format
          subject: lesson.subject,
          className: lesson.className,
          remedialDate: lesson.remedialDate, // This should be in YYYY-MM-DD format
          imageBase64: reader.result,
          gps: remedialGPSInput.value,
          duration: lesson.duration || 2,
          topic: lesson.topic || '',
          location: lesson.location || 'Classroom'
        };
        
        // Show loading
        showRemedialResult('<i class="fas fa-spinner fa-spin"></i> Submitting remedial attendance...', 'loading');
        
        google.script.run
          .withSuccessHandler(function(result) {
            if (result.success) {
              showRemedialResult(`
                <div class="remedial-success">
                  <i class="fas fa-check-circle"></i> ${result.message}
                  ${result.note ? `<br><small>${result.note}</small>` : ''}
                </div>
              `, 'success');
              
              // Clear form
              if (remedialGPSInput) remedialGPSInput.value = '';
              if (photoInput) photoInput.value = '';
              document.getElementById('selectedRemedialLesson').style.display = 'none';
              sessionStorage.removeItem('selectedRemedialLesson');
              
              // Refresh scheduled lessons after a delay
              setTimeout(() => {
                showScheduledLessons();
              }, 1500);
            } else {
              showRemedialResult(`
                <div class="remedial-error">
                  <i class="fas fa-exclamation-triangle"></i> ${result.message}
                </div>
              `, 'error');
            }
          })
          .withFailureHandler(function(error) {
            console.error('Server error:', error);
            showRemedialResult(`
              <div class="remedial-error">
                <i class="fas fa-exclamation-triangle"></i> Error: ${error.message || error}
              </div>
            `, 'error');
          })
          .submitRemedialAttendance(data);
      };
      
      reader.readAsDataURL(photoInput.files[0]);
    }

      };

      showRemedialResult('<i class="fas fa-spinner fa-spin"></i> Scheduling remedial lesson...', 'loading');

      google.script.run
        .withSuccessHandler(function(result) {

                <i class="fas fa-check-circle"></i> ${result.message}

                  <i class="fas fa-calendar-alt"></i> View Scheduled Lessons
                </button>
              </div>
            `, 'success');
            
            // Hide the scheduling section
            document.getElementById('remedialSchedulingSection').style.display = 'none';

              <i class="fas fa-exclamation-triangle"></i> Error: ${error.message || error}
            </div>
          `, 'error');
        })

    }
