    // Render teacher summary table
    function renderTeacherSummary(data) {
      const summaryDiv = document.getElementById('teacherSummary');
      
      if (!data || data.length === 0) {
        summaryDiv.innerHTML = `
          <div style="text-align: center; padding: 20px; color: #666;">
            <i class="fas fa-inbox fa-2x" style="margin-bottom: 10px;"></i>

      }
      
      // Calculate statistics
      const totalTeachers = data.length;
      const totalScheduled = data.reduce((sum, t) => sum + t.scheduledPeriods, 0);
      const totalAttended = data.reduce((sum, t) => sum + t.attendedPeriods, 0);
      const totalAbsent = data.reduce((sum, t) => sum + t.absentPeriods, 0);
      const totalUnattended = data.reduce((sum, t) => sum + t.unattendedPeriods, 0);
      const overallAttendance = totalScheduled > 0 ? Math.round((totalAttended / totalScheduled) * 100) : 0;
      
      // Get week range from first teacher's data
      const weekRange = data[0]?.weekRange || 'Current Week';
      
      let html = `
        <div style="margin-bottom: 15px; padding: 10px; background: #f8f9fa; border-radius: 8px;">
          <div style="display: flex; justify-content: space-between; align-items: center;">
            <div>
              <strong>Week:</strong> ${weekRange}
            </div>
            <div>
              <strong>Overall Attendance:</strong> 
              <span style="font-weight: bold; color: ${overallAttendance >= 80 ? '#27ae60' : overallAttendance >= 60 ? '#f39c12' : '#e74c3c'}">
                ${overallAttendance}%
              </span>
            </div>
          </div>
        </div>
        
        <div class="summary-stats">
          <div class="stat-badge">
            <span class="number">${totalTeachers}</span>
            <span class="label">Total Teachers</span>
          </div>
          <div class="stat-badge">
            <span class="number">${totalScheduled}</span>
            <span class="label">Scheduled Periods</span>
          </div>
          <div class="stat-badge" style="border-left: 4px solid #27ae60;">
            <span class="number">${totalAttended}</span>
            <span class="label">Attended Periods</span>
          </div>
          <div class="stat-badge" style="border-left: 4px solid #e74c3c;">
            <span class="number">${totalAbsent + totalUnattended}</span>
            <span class="label">Missed Periods</span>
          </div>
        </div>
        
        <div class="table-container" style="overflow-x: auto; max-height: 400px; overflow-y: auto;">
          <table class="summary-table">
            <thead>
              <tr>
                <th>Teacher Name</th>
                <th>Scheduled</th>
                <th>Present</th>
                <th>Absent</th>
                <th>No Record</th>
                <th>Attendance %</th>
                <th>Status</th>
              </tr>
            </thead>
            <tbody>
      `;
      
      data.forEach(teacher => {
        const percentage = teacher.attendancePercentage;
        let percentageClass = '';
        let status = '';
        let statusClass = '';
        
        if (percentage >= 90) {
          percentageClass = 'high-attendance';
          status = 'Excellent';
          statusClass = 'status-present';
        } else if (percentage >= 75) {
          percentageClass = 'medium-attendance';
          status = 'Good';
          statusClass = 'status-present';
        } else if (percentage >= 60) {
          percentageClass = 'medium-attendance';
          status = 'Fair';
          statusClass = 'status-present';
        } else {
          percentageClass = 'low-attendance';
          status = 'Needs Attention';
          statusClass = 'status-absent';
        }
        
        html += `
          <tr>
            <td><strong>${escapeHtml(teacher.name)}</strong></td>
            <td>${teacher.scheduledPeriods}</td>
            <td style="color: #27ae60; font-weight: bold;">${teacher.attendedPeriods}</td>
            <td style="color: #e74c3c;">${teacher.absentPeriods}</td>
            <td style="color: #f39c12;">${teacher.unattendedPeriods}</td>
            <td class="percentage-cell ${percentageClass}">${percentage}%</td>
            <td><span class="status-badge ${statusClass}">${status}</span></td>
          </tr>
        `;
      });
      
      html += `
            </tbody>

            <strong>Week:</strong> Based on Academic Year start date
          </div>
        </div>

      summaryDiv.innerHTML = html;

    // Prevent HTML injection errors
    function escapeHtml(text) {
      if (!text) return '';
      return text
        .toString()
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')
        .replace(/'/g, '&#039;');
    }

    // Render teachers list
    function renderTeachersList(teachers) {
      const teachersDiv = document.getElementById('teachersList');
      
      if (!teachers || teachers.length === 0) {
        teachersDiv.innerHTML = `
          <div style="text-align: center; padding: 40px; color: #666;">
            <i class="fas fa-users fa-3x" style="margin-bottom: 20px; opacity: 0.5;"></i>
            <h3>No Teachers Found</h3>
            <p>Add your first teacher to get started.</p>
            <button class="btn btn-success" onclick="showAddTeacherModal()">
              <i class="fas fa-user-plus"></i> Add First Teacher
            </button>
          </div>
        `;
        return;
      }
      
      let html = `
        <div class="table-container" style="overflow-x: auto;">
          <table class="data-table">
            <thead>
              <tr>
                <th>Name</th>
                <th>Email</th>
                <th>Phone</th>
                <th>Department</th>
                <th>Status</th>
                <th>Total Periods</th>
                <th>Teaching Load</th>
                <th>Actions</th>
              </tr>
            </thead>
            <tbody>
      `;
      
      teachers.forEach(teacher => {
        const statusClass = teacher.status === 'Active' ? 'status-present' : 
                          teacher.status === 'Inactive' ? 'status-absent' : 'status-badge';
        
        // Determine teaching load indicator
        let loadIndicator = '';
        let loadClass = '';
        if (teacher.totalPeriods >= 20) {
          loadIndicator = 'Heavy';
          loadClass = 'status-absent';
        } else if (teacher.totalPeriods >= 15) {
          loadIndicator = 'Moderate';
          loadClass = 'status-present';
        } else if (teacher.totalPeriods > 0) {
          loadIndicator = 'Light';
          loadClass = 'status-present';
        } else {
          loadIndicator = 'None';
          loadClass = '';
        }
        
        html += `
          <tr>
            <td><strong>${escapeHtml(teacher.name)}</strong></td>
            <td>${escapeHtml(teacher.email) || '-'}</td>
            <td>${escapeHtml(teacher.phone) || '-'}</td>
            <td>${escapeHtml(teacher.department) || '-'}</td>
            <td><span class="status-badge ${statusClass}">${teacher.status}</span></td>
            <td><strong>${teacher.totalPeriods}</strong></td>
            <td><span class="status-badge ${loadClass}">${loadIndicator}</span></td>
            <td>
              <button class="btn btn-primary btn-sm" onclick="viewTeacherDetails('${teacher.name}')" title="View Details">
                <i class="fas fa-eye"></i>
              </button>
              <button class="btn btn-warning btn-sm" onclick="editTeacher('${teacher.name}')" title="Edit">
                <i class="fas fa-edit"></i>
              </button>
              <button class="btn btn-danger btn-sm" onclick="deleteTeacherPrompt('${teacher.name}', ${teacher.rowIndex})" title="Delete">
                <i class="fas fa-trash"></i>
              </button>
            </td>
          </tr>
        `;
      });
      
      html += `</tbody></table></div>`;
      
      // Add summary statistics
      const totalPeriods = teachers.reduce((sum, teacher) => sum + teacher.totalPeriods, 0);
      const averagePeriods = teachers.length > 0 ? (totalPeriods / teachers.length).toFixed(1) : 0;
      
      html += `
        <div style="margin-top: 15px; padding: 15px; background: #f8f9fa; border-radius: 8px;">
          <strong>Summary Statistics:</strong> 
          ${teachers.length} teachers, 
          ${totalPeriods} total periods, 
          ${averagePeriods} average periods per teacher
        </div>
      `;
      
      teachersDiv.innerHTML = html;
    }

          allTeachers.forEach(teacher => {
            html += `<tr><td><strong>${escapeHtml(teacher.name)}</strong></td>`;

                html += `<span style="color: #999;">-</span>`;
              }
              
              html += `</td>`;
            }
            
            html += `</tr>`;
          });
          
          html += `</tbody></table></div>`;

    // Calculate and display period preview
    function calculatePeriodsPreview() {
      const startTime = document.getElementById('scheduleStartTime').value;
      const endTime = document.getElementById('scheduleEndTime').value;
      const conflictDiv = document.getElementById('scheduleConflictCheck');
      
      if (startTime && endTime) {
        // Calculate periods based on time difference
        const start = new Date(`1970-01-01T${startTime}`);
        const end = new Date(`1970-01-01T${endTime}`);
        
        let durationMinutes = (end - start) / (1000 * 60);
        
        // Handle overnight case (unlikely but possible)
        if (durationMinutes < 0) {
          durationMinutes += 24 * 60;
        }
        
        const periods = Math.floor(durationMinutes / 60);
        
        if (periods > 0) {
          conflictDiv.innerHTML = `
            <div class="success" style="padding: 10px;">
              <i class="fas fa-clock"></i> This schedule will be counted as <strong>${periods} period${periods !== 1 ? 's' : ''}</strong>
              (${durationMinutes} minutes total, ${durationMinutes - (periods * 60)} minutes break time)
            </div>
          `;
        } else {
          conflictDiv.innerHTML = `
            <div class="error" style="padding: 10px;">
              <i class="fas fa-exclamation-triangle"></i> Invalid time range: Must be at least 1 hour (60 minutes)
            </div>
          `;
        }
      } else {
        conflictDiv.innerHTML = '';
      }
    }

    function renderClassroomVisualization(classrooms) {
      const visualizationDiv = document.getElementById('classroomVisualization');
      
      if (!classrooms || classrooms.length === 0) {
        visualizationDiv.innerHTML = `
          <div class="no-classrooms">
            <i class="fas fa-school"></i>

            <p>Classroom information will appear here when available.</p>
          </div>
        `;
        return;
      }
      
      let filteredClassrooms = classrooms;
      if (currentFilter === 'occupied') {
        filteredClassrooms = classrooms.filter(c => c.status === 'occupied');
      } else if (currentFilter === 'vacant') {
        filteredClassrooms = classrooms.filter(c => c.status === 'vacant' && c.isCurrentPeriod);
      } else if (currentFilter === 'current') {
        filteredClassrooms = classrooms.filter(c => c.isCurrentPeriod);
      }
      
      if (filteredClassrooms.length === 0) {
        visualizationDiv.innerHTML = `
          <div class="no-classrooms">
            <i class="fas fa-search"></i>
            <h3>No Classrooms Match Filter</h3>
            <p>Try changing the filter to see different classrooms.</p>
          </div>
        `;
        return;
      }
      
      let html = '<div class="classroom-grid">';
      
      filteredClassrooms.forEach(classroom => {
        let statusClass = 'status-vacant';
        let cardClass = '';
        let statusText = 'Vacant';
        
        if (classroom.status === 'occupied') {
          statusClass = 'status-occupied';
          cardClass = 'occupied';
          statusText = 'Occupied';
        }
        
        if (classroom.isCurrentPeriod) {
          statusClass = 'status-current';
          cardClass += ' current-period';
          statusText += ' • Current Period';
        }
        
        html += `
          <div class="classroom-card ${cardClass.trim()}">
            <div class="classroom-header">
              <h4 class="classroom-name">
                <i class="fas ${classroom.status === 'occupied' ? 'chalkboard-teacher' : 'door-open'}"></i>

              </h4>
              <span class="classroom-status ${statusClass}">
                ${statusText}
              </span>
            </div>
            
            <div class="classroom-details">
        `;
        
        if (classroom.status === 'occupied') {
          html += `
            <div class="teacher-name">
              <i class="fas fa-user"></i> ${escapeHtml(classroom.teacher)}
            </div>
            <div class="subject-name">
              <i class="fas fa-book"></i> ${escapeHtml(classroom.subject)}

    // Set date range presets
    function setDateRange(rangeType) {
      const today = new Date();
      let startDate = new Date();
      let endDate = new Date();
      
      switch(rangeType) {
        case 'today':
          startDate = today;
          endDate = today;
          break;
        case 'yesterday':
          startDate = new Date(today);
          startDate.setDate(today.getDate() - 1);
          endDate = new Date(startDate);
          break;
        case 'thisWeek':
          startDate = new Date(today);
          startDate.setDate(today.getDate() - today.getDay() + 1); // Start of week (Monday)
          endDate = new Date(today);
          break;
        case 'lastWeek':
          startDate = new Date(today);
          startDate.setDate(today.getDate() - today.getDay() - 6); // Start of last week
          endDate = new Date(startDate);
          endDate.setDate(startDate.getDate() + 6); // End of last week
          break;
        case 'thisMonth':
          startDate = new Date(today.getFullYear(), today.getMonth(), 1);
          endDate = new Date(today.getFullYear(), today.getMonth() + 1, 0);
          break;
        case 'all':
          // Use a very old date and today
          startDate = new Date(2000, 0, 1);
          endDate = today;
          break;
      }
      
      document.getElementById('clearanceStartDate').value = formatDateForInput(startDate);
      document.getElementById('clearanceEndDate').value = formatDateForInput(endDate);
    }

    // Helper function to format dates
    function formatDateForDisplay(dateValue) {
      try {
        if (!dateValue) return 'N/A';
        
        // If it's already a Date object
        if (dateValue instanceof Date) {
          return dateValue.toISOString().split('T')[0]; // Return YYYY-MM-DD
        }
        
        // If it's a string, try to parse it
        if (typeof dateValue === 'string') {
          // Check if it's already in YYYY-MM-DD format
          if (dateValue.match(/^\d{4}-\d{2}-\d{2}$/)) {
            return dateValue;
          }
          
          // Try to parse as date
          const dateObj = new Date(dateValue);
          if (!isNaN(dateObj.getTime())) {
            return dateObj.toISOString().split('T')[0];
          }
          
          return dateValue; // Return as-is if can't parse
        }
        
        // For any other type, convert to string
        return dateValue.toString();
      } catch (e) {
        console.error('Error formatting date:', e);
        return dateValue ? dateValue.toString() : 'N/A';
      }
    }

// Set manual date presets
function setManualDate(rangeType) {
    const today = new Date();
    let targetDate = new Date();
    
    switch(rangeType) {
        case 'today':
            targetDate = today;
            break;
        case 'yesterday':
            targetDate = new Date(today);
            targetDate.setDate(today.getDate() - 1);
            break;
        case 'friday':
            // Get last Friday's date from server
            google.script.run
                .withSuccessHandler(function(dateStr) {
                    document.getElementById('manualAbsenceDate').value = dateStr;
                    logManualMessage(`Set date to last Friday: ${dateStr}`, 'info');
                })
                .withFailureHandler(function(error) {
                    logManualMessage('Error getting last Friday date: ' + error, 'error');
                })
                .getLastFriday();
            return;
        case 'monday':
            // Get last Monday's date from server
            google.script.run
                .withSuccessHandler(function(dateStr) {
                    document.getElementById('manualAbsenceDate').value = dateStr;
                    logManualMessage(`Set date to last Monday: ${dateStr}`, 'info');
                })
                .withFailureHandler(function(error) {
                    logManualMessage('Error getting last Monday date: ' + error, 'error');
                })
                .getLastMonday();
            return;
    }
    
    document.getElementById('manualAbsenceDate').value = formatDateForInput(targetDate);
}

// Log messages to the manual operation log
function logManualMessage(message, type = 'info') {
    const logDiv = document.getElementById('manualOperationLog');
    const timestamp = new Date().toLocaleTimeString();
    
    let icon = 'ℹ️';
    let color = '#666';
    
    switch(type) {
        case 'success':
            icon = '✅';
            color = '#27ae60';
            break;
        case 'error':
            icon = '❌';
            color = '#e74c3c';
            break;
        case 'warning':
            icon = '⚠️';
            color = '#f39c12';
            break;
        case 'info':
            icon = 'ℹ️';
            color = '#3498db';
            break;
    }
    
    const logEntry = document.createElement('div');
    logEntry.style.marginBottom = '5px';
    logEntry.style.color = color;
    logEntry.innerHTML = `<strong>[${timestamp}]</strong> ${icon} ${message}`;
    
    logDiv.appendChild(logEntry);
    
    // Scroll to bottom
    logDiv.scrollTop = logDiv.scrollHeight;
}
